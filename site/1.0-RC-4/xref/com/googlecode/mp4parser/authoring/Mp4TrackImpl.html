45      public Mp4TrackImpl(TrackBox trackBox) {
46          samples = new SampleList(trackBox);
47          SampleTableBox stbl = trackBox.getMediaBox().getMediaInformationBox().getSampleTableBox();
48          handler = trackBox.getMediaBox().getHandlerBox().getHandlerType();
49  
50  
51          mihd = trackBox.getMediaBox().getMediaInformationBox().getMediaHeaderBox();
52  
53  
54          sampleDescriptionBox = stbl.getSampleDescriptionBox();
55          if (trackBox.getParent().getBoxes(MovieExtendsBox.class).size() > 0) {
56  
57              decodingTimeEntries = new LinkedList<TimeToSampleBox.Entry>();
58              compositionTimeEntries = new LinkedList<CompositionTimeToSample.Entry>();
59              sampleDependencies = new LinkedList<SampleDependencyTypeBox.Entry>();
60  
61              for (MovieFragmentBox movieFragmentBox : trackBox.getIsoFile().getBoxes(MovieFragmentBox.class)) {
62                  List<TrackFragmentBox> trafs = movieFragmentBox.getBoxes(TrackFragmentBox.class);
63                  for (TrackFragmentBox traf : trafs) {
64                      if (traf.getTrackFragmentHeaderBox().getTrackId() == trackBox.getTrackHeaderBox().getTrackId()) {
65                          List<TrackRunBox> truns = traf.getBoxes(TrackRunBox.class);
66                          for (TrackRunBox trun : truns) {
67                              for (TrackRunBox.Entry entry : trun.getEntries()) {
68                                  if (trun.isSampleDurationPresent()) {
69                                      if (decodingTimeEntries.size() == 0 ||
70                                              decodingTimeEntries.get(decodingTimeEntries.size() - 1).getDelta() != entry.getSampleDuration()) {
71                                          decodingTimeEntries.add(new TimeToSampleBox.Entry(1, entry.getSampleDuration()));
72                                      } else {
73                                          TimeToSampleBox.Entry e = decodingTimeEntries.get(decodingTimeEntries.size() - 1);
74                                          e.setCount(e.getCount() + 1);
75                                      }
76                                  }
77                                  if (trun.isSampleCompositionTimeOffsetPresent()) {
78                                      if (compositionTimeEntries.size() == 0 ||
79                                              compositionTimeEntries.get(compositionTimeEntries.size() - 1).getOffset() != entry.getSampleCompositionTimeOffset()) {
80                                          compositionTimeEntries.add(new CompositionTimeToSample.Entry(1, l2i(entry.getSampleCompositionTimeOffset())));
81                                      } else {
82                                          CompositionTimeToSample.Entry e = compositionTimeEntries.get(compositionTimeEntries.size() - 1);
83                                          e.setCount(e.getCount() + 1);
84                                      }
85                                  }
86  
87                              }
88  
89  
90                          }
91  
92  
93                      }
94                  }
95              }
96          } else {
97              decodingTimeEntries = stbl.getTimeToSampleBox().getEntries();
98              if (stbl.getCompositionTimeToSample() != null) {
99                  compositionTimeEntries = stbl.getCompositionTimeToSample().getEntries();
100             }
101             if (stbl.getSyncSampleBox() != null) {
102                 syncSamples = stbl.getSyncSampleBox().getSampleNumber();
103             }
104             if (stbl.getSampleDependencyTypeBox() != null) {
105                 sampleDependencies = stbl.getSampleDependencyTypeBox().getEntries();
106             }
107         }
108         MediaHeaderBox mdhd = trackBox.getMediaBox().getMediaHeaderBox();
109         TrackHeaderBox tkhd = trackBox.getTrackHeaderBox();
110 
111         setEnabled(tkhd.isEnabled());
112         setInMovie(tkhd.isInMovie());
113         setInPoster(tkhd.isInPoster());
114         setInPreview(tkhd.isInPreview());
115 
116         trackMetaData.setTrackId(tkhd.getTrackId());
117         trackMetaData.setCreationTime(DateHelper.convert(mdhd.getCreationTime()));
118         trackMetaData.setLanguage(mdhd.getLanguage());
119 /*        System.err.println(mdhd.getModificationTime());

121         System.err.println(DateHelper.convert(DateHelper.convert(mdhd.getModificationTime())));
122         System.err.println(DateHelper.convert(DateHelper.convert(DateHelper.convert(mdhd.getModificationTime()))));*/

124         trackMetaData.setModificationTime(DateHelper.convert(mdhd.getModificationTime()));
125         trackMetaData.setTimescale(mdhd.getTimescale());
126         trackMetaData.setHeight(tkhd.getHeight());
127         trackMetaData.setWidth(tkhd.getWidth());
