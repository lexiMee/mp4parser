37  /**
38   * A basic on-demand parsing box. Requires the implementation of three methods to become a fully working box:
39   * <ol>
40   * <li>{@link #_parseDetails(java.nio.ByteBuffer)}</li>
41   * <li>{@link #getContent(java.nio.ByteBuffer)}</li>
42   * <li>{@link #getContentSize()}</li>
43   * </ol>
44   * additionally this new box has to be put into the <code>isoparser-default.properties</code> file so that
45   * it is accessible by the <code>PropertyBoxParserImpl</code>
46   */
47  public abstract class AbstractBox implements Box {
48      private static Logger LOG = Logger.getLogger(AbstractBox.class.getName());
49  
50      protected String type;
51      private byte[] userType;
52      private ContainerBox parent;
53  
54      private ByteBuffer content;
55      private ByteBuffer deadBytes = null;
56  
57  
58      protected AbstractBox(String type) {
59          this.type = type;
60      }
61  
62      protected AbstractBox(String type, byte[] userType) {
63          this.type = type;
64          this.userType = userType;
65      }

67      /**
68       * Get the box's content size without its header. This must be the exact number of bytes
69       * that <code>getContent(ByteBuffer)</code> writes.
70       *
71       * @return Gets the box's content size in bytes
72       * @see #getContent(java.nio.ByteBuffer)
73       */
74      protected abstract long getContentSize();
75  
76      /**
77       * Write the box's content into the given <code>ByteBuffer</code>. This must include flags
78       * and version in case of a full box. <code>byteBuffer</code> has been initialized with
79       * <code>getSize()</code> bytes.
80       *
81       * @param byteBuffer the sink for the box's content
82       */
83      protected abstract void getContent(ByteBuffer byteBuffer);
84  
85      /**
86       * Parse the box's fields and child boxes if any.
87       *
88       * @param content the box's raw content beginning after the 4-cc field.
89       */
90      protected abstract void _parseDetails(ByteBuffer content);
91  
92      /**
93       * Read the box's content from a byte channel without parsing it. Parsing is done on-demand.
94       *
95       * @param readableByteChannel the (part of the) iso file to parse
96       * @param contentSize         expected contentSize of the box
97       * @param boxParser           creates inner boxes
98       * @throws IOException in case of an I/O error.
99       */
100     @DoNotParseDetail
101     public void parse(ReadableByteChannel readableByteChannel, ByteBuffer header, long contentSize, BoxParser boxParser) throws IOException {
102         if (readableByteChannel instanceof FileChannel && contentSize > 1024 * 1024) {
103             // It's quite expensive to map a file into the memory. Just do it when the box is larger than a MB.
104             content = ((FileChannel) readableByteChannel).map(FileChannel.MapMode.READ_ONLY, ((FileChannel) readableByteChannel).position(), contentSize);
105             ((FileChannel) readableByteChannel).position(((FileChannel) readableByteChannel).position() + contentSize);
106         } else {
107             assert contentSize < Integer.MAX_VALUE;
108             content = ChannelHelper.readFully(readableByteChannel, contentSize);
109         }
110     }

112     public void getBox(WritableByteChannel os) throws IOException {
113         ByteBuffer bb = ByteBuffer.allocate(l2i(getSize()));
114         getHeader(bb);
115         if (content == null) {
116             getContent(bb);
117             if (deadBytes != null) {
118                 deadBytes.rewind();
119                 while (deadBytes.remaining() > 0) {
120                     bb.put(deadBytes);
121                 }
122             }
123         } else {
124             content.rewind();
125             bb.put(content);
126         }
127         bb.rewind();
128         os.write(bb);
129     }

132     /**
133      * Parses the raw content of the box. It surrounds the actual parsing
134      * which is done
135      */
136     synchronized final void parseDetails() {
137         if (content != null) {
138             ByteBuffer content = this.content;
139             this.content = null;
140             content.rewind();
141             _parseDetails(content);
142             if (content.remaining() > 0) {
143                 deadBytes = content.slice();
144             }
145             assert verify(content);
146         }
147     }

149     /**
150      * Sets the 'dead' bytes. These bytes are left if the content of the box
151      * has been parsed but not all bytes have been used up.
152      *
153      * @param newDeadBytes the unused bytes with no meaning but required for bytewise reconstruction
154      */
155     protected void setDeadBytes(ByteBuffer newDeadBytes) {
156         deadBytes = newDeadBytes;
157     }

160     /**
161      * Gets the full size of the box including header and content.
162      *
163      * @return the box's size
164      */
165     public long getSize() {
166         long size = (content == null ? getContentSize() : content.limit());
167         size += (8 + // size|type
168                 (size >= ((1L << 32) - 8) ? 8 : 0) + // 32bit - 8 byte size and type
169                 (UserBox.TYPE.equals(getType()) ? 16 : 0));
170         size += (deadBytes == null ? 0 : deadBytes.limit());
171         return size;
172     }

209     /**
210      * Verifies that a box can be reconstructed byte-exact after parsing.
211      *
212      * @param content the raw content of the box
213      * @return <code>true</code> if raw content exactly matches the reconstructed content
214      */
215     private boolean verify(ByteBuffer content) {
216         ByteBuffer bb = ByteBuffer.allocate(l2i(getContentSize() + (deadBytes != null ? deadBytes.limit() : 0)));
217         getContent(bb);
218         if (deadBytes != null) {
219             deadBytes.rewind();
220             while (deadBytes.remaining() > 0) {
221                 bb.put(deadBytes);
222             }
223         }
224         content.rewind();
225         bb.rewind();
226 
227 
228         if (content.remaining() != bb.remaining()) {
229             LOG.severe("remaining differs " + content.remaining() + " vs. " + bb.remaining());
230             return false;
231         }
232         int p = content.position();
233         for (int i = content.limit() - 1, j = bb.limit() - 1; i >= p; i--, j--) {
234             byte v1 = content.get(i);
235             byte v2 = bb.get(j);
236             if (v1 != v2) {
237                 LOG.severe("buffers differ at " + i + "/" + j);
238                 return false;
239             }
240         }
241         return true;
242 
243     }

245     private boolean isSmallBox() {
246         return (content == null ? (getContentSize() + (deadBytes != null ? deadBytes.limit() : 0) + 8) : content.limit()) < 1L << 32;
247     }
248 
249     private void getHeader(ByteBuffer byteBuffer) {
250         if (isSmallBox()) {
251             IsoTypeWriter.writeUInt32(byteBuffer, this.getSize());
252             byteBuffer.put(IsoFile.fourCCtoBytes(getType()));
253         } else {
254             IsoTypeWriter.writeUInt32(byteBuffer, 1);
255             byteBuffer.put(IsoFile.fourCCtoBytes(getType()));
256             IsoTypeWriter.writeUInt64(byteBuffer, getSize());
257         }
258         if (UserBox.TYPE.equals(getType())) {
259             byteBuffer.put(getUserType());
260         }
