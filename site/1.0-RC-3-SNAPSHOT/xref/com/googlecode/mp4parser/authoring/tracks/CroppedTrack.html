27  /**
28   * Generates a Track that starts at fromSample and ends at toSample (exclusive). The user of this class
29   * has to make sure that the fromSample is a random access sample.
30   * <ul>
31   * <li>In AAC this is every single sample</li>
32   * <li>In H264 this is every sample that is marked in the SyncSampleBox</li>
33   * </ul>
34   */
35  public class CroppedTrack extends AbstractTrack {
36      Track origTrack;
37      private int fromSample;
38      private int toSample;
39      private long[] syncSampleArray;
40  
41      public CroppedTrack(Track origTrack, long fromSample, long toSample) {
42          this.origTrack = origTrack;
43          assert fromSample <= Integer.MAX_VALUE;
44          assert toSample <= Integer.MAX_VALUE;
45          this.fromSample = (int) fromSample;
46          this.toSample = (int) toSample;
47      }
48  
49      public List<ByteBuffer> getSamples() {
50          return origTrack.getSamples().subList(fromSample, toSample);
51      }
52  
53      public SampleDescriptionBox getSampleDescriptionBox() {
54          return origTrack.getSampleDescriptionBox();
55      }

57      public List<TimeToSampleBox.Entry> getDecodingTimeEntries() {
58          if (origTrack.getDecodingTimeEntries() != null && !origTrack.getDecodingTimeEntries().isEmpty()) {
59              long[] decodingTimes = TimeToSampleBox.blowupTimeToSamples(origTrack.getDecodingTimeEntries());
60              long[] nuDecodingTimes = new long[toSample - fromSample];
61              System.arraycopy(decodingTimes, fromSample, nuDecodingTimes, 0, toSample - fromSample);
62  
63              LinkedList<TimeToSampleBox.Entry> returnDecodingEntries = new LinkedList<TimeToSampleBox.Entry>();
64  
65              for (long nuDecodingTime : nuDecodingTimes) {
66                  if (returnDecodingEntries.isEmpty() || returnDecodingEntries.getLast().getDelta() != nuDecodingTime) {
67                      TimeToSampleBox.Entry e = new TimeToSampleBox.Entry(1, nuDecodingTime);
68                      returnDecodingEntries.add(e);
69                  } else {
70                      TimeToSampleBox.Entry e = returnDecodingEntries.getLast();
71                      e.setCount(e.getCount() + 1);
72                  }
73              }
74              return returnDecodingEntries;
75          } else {
76              return null;
77          }
78      }
79  
80      public List<CompositionTimeToSample.Entry> getCompositionTimeEntries() {
81          if (origTrack.getCompositionTimeEntries() != null && !origTrack.getCompositionTimeEntries().isEmpty()) {
82              int[] compositionTime = CompositionTimeToSample.blowupCompositionTimes(origTrack.getCompositionTimeEntries());
83              int[] nuCompositionTimes = new int[toSample - fromSample];
84              System.arraycopy(compositionTime, fromSample, nuCompositionTimes, 0, toSample - fromSample);
85  
86              LinkedList<CompositionTimeToSample.Entry> returnDecodingEntries = new LinkedList<CompositionTimeToSample.Entry>();
87  
88              for (int nuDecodingTime : nuCompositionTimes) {
89                  if (returnDecodingEntries.isEmpty() || returnDecodingEntries.getLast().getOffset() != nuDecodingTime) {
90                      CompositionTimeToSample.Entry e = new CompositionTimeToSample.Entry(1, nuDecodingTime);
91                      returnDecodingEntries.add(e);
92                  } else {
93                      CompositionTimeToSample.Entry e = returnDecodingEntries.getLast();
94                      e.setCount(e.getCount() + 1);
95                  }
96              }
97              return returnDecodingEntries;
98          } else {
99              return null;
100         }
101     }

103     synchronized public long[] getSyncSamples() {
104         if (this.syncSampleArray == null) {
105             if (origTrack.getSyncSamples() != null && origTrack.getSyncSamples().length > 0) {
106                 List<Long> syncSamples = new LinkedList<Long>();
107                 for (long l : origTrack.getSyncSamples()) {
108                     if (l >= fromSample && l < toSample) {
109                         syncSamples.add(l - fromSample);
110                     }
111                 }
112                 syncSampleArray = new long[syncSamples.size()];
113                 for (int i = 0; i < syncSampleArray.length; i++) {
114                     syncSampleArray[i] = syncSamples.get(i);
115 
116                 }
117                 return syncSampleArray;
118             } else {
119                 return null;
120             }
121         } else {
122             return this.syncSampleArray;
123         }
124     }
