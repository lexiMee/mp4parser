89      public IsoFile build(Movie movie) throws IOException {
90          LOG.fine("Creating movie " + movie);
91          for (Track track : movie.getTracks()) {
92              // getting the samples may be a time consuming activity
93              List<ByteBuffer> samples = track.getSamples();
94              track2Sample.put(track, samples);
95              long[] sizes = new long[samples.size()];
96              for (int i = 0; i < sizes.length; i++) {
97                  sizes[i] = samples.get(i).limit();
98              }
99              track2SampleSizes.put(track, sizes);
100         }
101 
102         IsoFile isoFile = new IsoFile();
103         // ouch that is ugly but I don't know how to do it else
104         List<String> minorBrands = new LinkedList<String>();
105         minorBrands.add("isom");
106         minorBrands.add("iso2");
107         minorBrands.add("avc1");
108 
109         isoFile.addBox(new FileTypeBox("isom", 0, minorBrands));
110         isoFile.addBox(createMovieBox(movie));
111         InterleaveChunkMdat mdat = new InterleaveChunkMdat(movie);
112         isoFile.addBox(mdat);
113 
114         /*
115         dataOffset is where the first sample starts. In this special mdat the samples always start
116         at offset 16 so that we can use the same offset for large boxes and small boxes
117          */
118         long dataOffset = mdat.getDataOffset();
119         for (StaticChunkOffsetBox chunkOffsetBox : chunkOffsetBoxes) {
120             long[] offsets = chunkOffsetBox.getChunkOffsets();
121             for (int i = 0; i < offsets.length; i++) {
122                 offsets[i] += dataOffset;
123             }
124         }
125 
126 
127         return isoFile;
128     }

130     private MovieBox createMovieBox(Movie movie) {
131         MovieBox movieBox = new MovieBox();
132         List<Box> movieBoxChildren = new LinkedList<Box>();
133         MovieHeaderBox mvhd = new MovieHeaderBox();
134         mvhd.setCreationTime(DateHelper.convert(new Date()));
135         mvhd.setModificationTime(DateHelper.convert(new Date()));
136 
137         long movieTimeScale = getTimescale(movie);
138         long duration = 0;
139 
140         for (Track track : movie.getTracks()) {
141             long tracksDuration = getDuration(track) * movieTimeScale / track.getTrackMetaData().getTimescale();
142             if (tracksDuration > duration) {
143                 duration = tracksDuration;
144             }
145 
146 
147         }
148 
149         mvhd.setDuration(duration);
150         mvhd.setTimescale(movieTimeScale);
151         // find the next available trackId
152         long nextTrackId = 0;
153         for (Track track : movie.getTracks()) {
154             nextTrackId = nextTrackId < track.getTrackMetaData().getTrackId() ? track.getTrackMetaData().getTrackId() : nextTrackId;
155         }
156         mvhd.setNextTrackId(++nextTrackId);
157         movieBoxChildren.add(mvhd);
158         for (Track track : movie.getTracks()) {
159             movieBoxChildren.add(createTrackBox(track, movie));
160         }
161         // metadata here
162         movieBox.setBoxes(movieBoxChildren);
163         return movieBox;
164 
165     }

167     private TrackBox createTrackBox(Track track, Movie movie) {
168 
169         LOG.info("Creating Mp4TrackImpl " + track);
170         TrackBox trackBox = new TrackBox();
171         TrackHeaderBox tkhd = new TrackHeaderBox();
172         int flags = 0;
173         if (track.isEnabled()) {
174             flags += 1;
175         }
176 
177         if (track.isInMovie()) {
178             flags += 2;
179         }
180 
181         if (track.isInPreview()) {
182             flags += 4;
183         }
184 
185         if (track.isInPoster()) {
186             flags += 8;
187         }
188         tkhd.setFlags(flags);
189 
190         tkhd.setAlternateGroup(track.getTrackMetaData().getGroup());
191         tkhd.setCreationTime(DateHelper.convert(track.getTrackMetaData().getCreationTime()));
192         // We need to take edit list box into account in trackheader duration
193         // but as long as I don't support edit list boxes it is sufficient to
194         // just translate media duration to movie timescale
195         tkhd.setDuration(getDuration(track) * getTimescale(movie) / track.getTrackMetaData().getTimescale());
196         tkhd.setHeight(track.getTrackMetaData().getHeight());
197         tkhd.setWidth(track.getTrackMetaData().getWidth());
198         tkhd.setLayer(track.getTrackMetaData().getLayer());
199         tkhd.setModificationTime(DateHelper.convert(new Date()));
200         tkhd.setTrackId(track.getTrackMetaData().getTrackId());
201         tkhd.setVolume(track.getTrackMetaData().getVolume());
202         trackBox.addBox(tkhd);
203 
204         EditBox edit = new EditBox();
205         EditListBox editListBox = new EditListBox();
206         editListBox.setEntries(Collections.singletonList(
207                 new EditListBox.Entry(editListBox, (long) (track.getTrackMetaData().getStartTime() * getTimescale(movie)), -1, 1)));
208         edit.addBox(editListBox);
209         trackBox.addBox(edit);
210 
211         MediaBox mdia = new MediaBox();
212         trackBox.addBox(mdia);
213         MediaHeaderBox mdhd = new MediaHeaderBox();
214         mdhd.setCreationTime(DateHelper.convert(track.getTrackMetaData().getCreationTime()));
215         mdhd.setDuration(getDuration(track));
216         mdhd.setTimescale(track.getTrackMetaData().getTimescale());
217         mdhd.setLanguage(track.getTrackMetaData().getLanguage());
218         mdia.addBox(mdhd);
219         HandlerBox hdlr = new HandlerBox();
220         mdia.addBox(hdlr);
221 
222         hdlr.setHandlerType(track.getHandler());
223 
224         MediaInformationBox minf = new MediaInformationBox();
225         minf.addBox(track.getMediaHeaderBox());
226 
227         // dinf: all these three boxes tell us is that the actual
228         // data is in the current file and not somewhere external
229         DataInformationBox dinf = new DataInformationBox();
230         DataReferenceBox dref = new DataReferenceBox();
231         dinf.addBox(dref);
232         DataEntryUrlBox url = new DataEntryUrlBox();
233         url.setFlags(1);
234         dref.addBox(url);
235         minf.addBox(dinf);
236         //
237 
238         SampleTableBox stbl = new SampleTableBox();
239 
240         stbl.addBox(track.getSampleDescriptionBox());
241 
242         List<TimeToSampleBox.Entry> decodingTimeToSampleEntries = track.getDecodingTimeEntries();
243         if (decodingTimeToSampleEntries != null && !track.getDecodingTimeEntries().isEmpty()) {
244             TimeToSampleBox stts = new TimeToSampleBox();
245             stts.setEntries(track.getDecodingTimeEntries());
246             stbl.addBox(stts);
247         }
248 
249         List<CompositionTimeToSample.Entry> compositionTimeToSampleEntries = track.getCompositionTimeEntries();
250         if (compositionTimeToSampleEntries != null && !compositionTimeToSampleEntries.isEmpty()) {
251             CompositionTimeToSample ctts = new CompositionTimeToSample();
252             ctts.setEntries(compositionTimeToSampleEntries);
253             stbl.addBox(ctts);
254         }
255 
256         long[] syncSamples = track.getSyncSamples();
257         if (syncSamples != null && syncSamples.length > 0) {
258             SyncSampleBox stss = new SyncSampleBox();
259             stss.setSampleNumber(syncSamples);
260             stbl.addBox(stss);
261         }
262 
263         if (track.getSampleDependencies() != null && !track.getSampleDependencies().isEmpty()) {
264             SampleDependencyTypeBox sdtp = new SampleDependencyTypeBox();
265             sdtp.setEntries(track.getSampleDependencies());
266             stbl.addBox(sdtp);
267         }
268         int chunkSize[] = getChunkSizes(track, movie);
269         SampleToChunkBox stsc = new SampleToChunkBox();
270         stsc.setEntries(new LinkedList<SampleToChunkBox.Entry>());
271         long lastChunkSize = Integer.MIN_VALUE; // to be sure the first chunks hasn't got the same size
272         for (int i = 0; i < chunkSize.length; i++) {
273             // The sample description index references the sample description box
274             // that describes the samples of this chunk. My Tracks cannot have more
275             // than one sample description box. Therefore 1 is always right
276             // the first chunk has the number '1'
277             if (lastChunkSize != chunkSize[i]) {
278                 stsc.getEntries().add(new SampleToChunkBox.Entry(i + 1, chunkSize[i], 1));
279                 lastChunkSize = chunkSize[i];
280             }
281         }
282         stbl.addBox(stsc);
283 
284         SampleSizeBox stsz = new SampleSizeBox();
285         stsz.setSampleSizes(track2SampleSizes.get(track));
286 
287         stbl.addBox(stsz);
288         // The ChunkOffsetBox we create here is just a stub
289         // since we haven't created the whole structure we can't tell where the
290         // first chunk starts (mdat box). So I just let the chunk offset
291         // start at zero and I will add the mdat offset later.
292         StaticChunkOffsetBox stco = new StaticChunkOffsetBox();
293         this.chunkOffsetBoxes.add(stco);
294         long offset = 0;
295         long[] chunkOffset = new long[chunkSize.length];
296         // all tracks have the same number of chunks
297         LOG.fine("Calculating chunk offsets for track_" + track.getTrackMetaData().getTrackId());
298         for (int i = 0; i < chunkSize.length; i++) {
299             // The filelayout will be:
300             // chunk_1_track_1,... ,chunk_1_track_n, chunk_2_track_1,... ,chunk_2_track_n, ... , chunk_m_track_1,... ,chunk_m_track_n
301             // calculating the offsets
302             LOG.finer("Calculating chunk offsets for track_" + track.getTrackMetaData().getTrackId() + " chunk " + i);
303             for (Track current : movie.getTracks()) {
304                 LOG.finest("Adding offsets of track_" + current.getTrackMetaData().getTrackId());
305                 int[] chunkSizes = getChunkSizes(current, movie);
306                 long firstSampleOfChunk = 0;
307                 for (int j = 0; j < i; j++) {
308                     firstSampleOfChunk += chunkSizes[j];
309                 }
310                 if (current == track) {
311                     chunkOffset[i] = offset;
312                 }
313                 for (int j = l2i(firstSampleOfChunk); j < firstSampleOfChunk + chunkSizes[i]; j++) {
314                     offset += track2SampleSizes.get(current)[j];
315                 }
316             }
317         }
318         stco.setChunkOffsets(chunkOffset);
319         stbl.addBox(stco);
320         minf.addBox(stbl);
321         mdia.addBox(minf);
322 
323         return trackBox;
324     }

344         private InterleaveChunkMdat(Movie movie) {
345 
346             tracks = movie.getTracks();
347             Map<Track, int[]> chunks = new HashMap<Track, int[]>();
348             for (Track track : movie.getTracks()) {
349                 chunks.put(track, getChunkSizes(track, movie));
350             }
351 
352             for (int i = 0; i < chunks.values().iterator().next().length; i++) {
353                 for (Track track : tracks) {
354 
355                     int[] chunkSizes = chunks.get(track);
356                     long firstSampleOfChunk = 0;
357                     for (int j = 0; j < i; j++) {
358                         firstSampleOfChunk += chunkSizes[j];
359                     }
360 
361                     for (int j = l2i(firstSampleOfChunk); j < firstSampleOfChunk + chunkSizes[i]; j++) {
362 
363                         ByteBuffer s = DefaultMp4Builder.this.track2Sample.get(track).get(j);
364                         contentSize += s.limit();
365                         samples.add((ByteBuffer) s.rewind());
366                     }
367 
368                 }
369 
370             }
371 
372         }

374         public long getDataOffset() {
375             Box b = this;
376             long offset = 16;
377             while (b.getParent() != null) {
378                 for (Box box : b.getParent().getBoxes()) {
379                     if (b == box) {
380                         break;
381                     }
382                     offset += box.getSize();
383                 }
384                 b = b.getParent();
385             }
386             return offset;
387         }

403         public void getBox(WritableByteChannel writableByteChannel) throws IOException {
404             ByteBuffer bb = ByteBuffer.allocate(16);
405             long size = getSize();
406             if (isSmallBox(size)) {
407                 IsoTypeWriter.writeUInt32(bb, size);
408             } else {
409                 IsoTypeWriter.writeUInt32(bb, 1);
410             }
411             bb.put(IsoFile.fourCCtoBytes("mdat"));
412             if (isSmallBox(size)) {
413                 bb.put(new byte[8]);
414             } else {
415                 IsoTypeWriter.writeUInt64(bb, size);
416             }
417             bb.rewind();
418             writableByteChannel.write(bb);
419             if (writableByteChannel instanceof GatheringByteChannel) {
420                 List<ByteBuffer> nuSamples = unifyAdjacentBuffers(samples);
421 
422                 int STEPSIZE = 1024;
423                 for (int i = 0; i < Math.ceil((double) nuSamples.size() / STEPSIZE); i++) {
424                     List<ByteBuffer> sublist = nuSamples.subList(
425                             i * STEPSIZE, // start
426                             (i + 1) * STEPSIZE < nuSamples.size() ? (i + 1) * STEPSIZE : nuSamples.size()); // end
427                     ByteBuffer sampleArray[] = sublist.toArray(new ByteBuffer[sublist.size()]);
428                     do {
429                         ((GatheringByteChannel) writableByteChannel).write(sampleArray);
430                     } while (sampleArray[sampleArray.length - 1].remaining() > 0);
431                 }
432                 //System.err.println(bytesWritten);
433             } else {
434                 for (ByteBuffer sample : samples) {
435                     sample.rewind();
436                     writableByteChannel.write(sample);
437                 }
438             }
439         }

443     /**
444      * Gets the chunk sizes for the given track.
445      *
446      * @param track
447      * @param movie
448      * @return
449      */
450     int[] getChunkSizes(Track track, Movie movie) {
451 
452         long[] referenceChunkStarts = intersectionFinder.sampleNumbers(track, movie);
453         int[] chunkSizes = new int[referenceChunkStarts.length];
454 
455 
456         for (int i = 0; i < referenceChunkStarts.length; i++) {
457             long start = referenceChunkStarts[i] - 1;
458             long end;
459             if (referenceChunkStarts.length == i + 1) {
460                 end = track.getSamples().size() - 1;
461             } else {
462                 end = referenceChunkStarts[i + 1] - 1;
463             }
464 
465             chunkSizes[i] = l2i(end - start);
466             // The Stretch makes sure that there are as much audio and video chunks!
467         }
468         assert DefaultMp4Builder.this.track2Sample.get(track).size() == sum(chunkSizes) : "The number of samples and the sum of all chunk lengths must be equal";
469         return chunkSizes;
470 
471

506     public List<ByteBuffer> unifyAdjacentBuffers(List<ByteBuffer> samples) {
507         ArrayList<ByteBuffer> nuSamples = new ArrayList<ByteBuffer>(samples.size());
508         for (ByteBuffer buffer : samples) {
509             int lastIndex = nuSamples.size() - 1;
510             if (lastIndex >= 0 && buffer.hasArray() && nuSamples.get(lastIndex).hasArray() && buffer.array() == nuSamples.get(lastIndex).array() &&
511                     nuSamples.get(lastIndex).arrayOffset() + nuSamples.get(lastIndex).limit() == buffer.arrayOffset()) {
512                 ByteBuffer oldBuffer = nuSamples.remove(lastIndex);
513                 ByteBuffer nu = ByteBuffer.wrap(buffer.array(), oldBuffer.arrayOffset(), oldBuffer.limit() + buffer.limit()).slice();
514                 // We need to slice here since wrap([], offset, length) just sets position and not the arrayOffset.
515                 nuSamples.add(nu);
516             } else if (lastIndex >= 0 &&
517                     buffer instanceof MappedByteBuffer && nuSamples.get(lastIndex) instanceof MappedByteBuffer &&
518                     nuSamples.get(lastIndex).limit() == nuSamples.get(lastIndex).capacity() - buffer.capacity()) {
519                 // This can go wrong - but will it?
520                 ByteBuffer oldBuffer = nuSamples.get(lastIndex);
521                 oldBuffer.limit(buffer.limit() + oldBuffer.limit());
522             } else {
523                 nuSamples.add(buffer);
524             }
525         }
526         return nuSamples;
