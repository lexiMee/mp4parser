57      public IsoFile build(Movie movie) {
58          LOG.fine("Creating movie " + movie);
59          for (Track track : movie.getTracks()) {
60              // getting the samples may be a time consuming activity
61              List<ByteBuffer> samples = track.getSamples();
62              putSamples(track, samples);
63              long[] sizes = new long[samples.size()];
64              for (int i = 0; i < sizes.length; i++) {
65                  ByteBuffer b = samples.get(i);
66                  sizes[i] = b.limit() - b.position();
67              }
68              track2SampleSizes.put(track, sizes);
69  
70          }
71  
72          IsoFile isoFile = new IsoFile();
73  
74          isoFile.addBox(createFileTypeBox(movie));
75  
76          Map<Track, int[]> chunks = new HashMap<Track, int[]>();
77          for (Track track : movie.getTracks()) {
78              chunks.put(track, getChunkSizes(track, movie));
79          }
80  
81          isoFile.addBox(createMovieBox(movie, chunks));
82          List<Box> stszs = Path.getPaths(isoFile, "/moov/trak/mdia/minf/stbl/stsz");
83  
84          long contentSize = 0;
85          for (Box stsz : stszs) {
86              contentSize += sum(((SampleSizeBox) stsz).getSampleSizes());
87  
88          }
89  
90          InterleaveChunkMdat mdat = new InterleaveChunkMdat(movie, chunks, contentSize);
91          isoFile.addBox(mdat);
92  
93          /*
94          dataOffset is where the first sample starts. In this special mdat the samples always start
95          at offset 16 so that we can use the same offset for large boxes and small boxes
96           */
97          long dataOffset = mdat.getDataOffset();
98          for (StaticChunkOffsetBox chunkOffsetBox : chunkOffsetBoxes) {
99              long[] offsets = chunkOffsetBox.getChunkOffsets();
100             for (int i = 0; i < offsets.length; i++) {
101                 offsets[i] += dataOffset;
102             }
103         }
104 
105 
106         return isoFile;
107     }

123     protected MovieBox createMovieBox(Movie movie, Map<Track, int[]> chunks) {
124         MovieBox movieBox = new MovieBox();
125         MovieHeaderBox mvhd = new MovieHeaderBox();
126 
127         mvhd.setCreationTime(DateHelper.convert(new Date()));
128         mvhd.setModificationTime(DateHelper.convert(new Date()));
129 
130         long movieTimeScale = getTimescale(movie);
131         long duration = 0;
132 
133         for (Track track : movie.getTracks()) {
134             long tracksDuration = getDuration(track) * movieTimeScale / track.getTrackMetaData().getTimescale();
135             if (tracksDuration > duration) {
136                 duration = tracksDuration;
137             }
138 
139 
140         }
141 
142         mvhd.setDuration(duration);
143         mvhd.setTimescale(movieTimeScale);
144         // find the next available trackId
145         long nextTrackId = 0;
146         for (Track track : movie.getTracks()) {
147             nextTrackId = nextTrackId < track.getTrackMetaData().getTrackId() ? track.getTrackMetaData().getTrackId() : nextTrackId;
148         }
149         mvhd.setNextTrackId(++nextTrackId);
150         if (mvhd.getCreationTime() >= 1l << 32 ||
151                 mvhd.getModificationTime() >= 1l << 32 ||
152                 mvhd.getDuration() >= 1l << 32) {
153             mvhd.setVersion(1);
154         }
155 
156         movieBox.addBox(mvhd);
157         for (Track track : movie.getTracks()) {
158             movieBox.addBox(createTrackBox(track, movie, chunks));
159         }
160         // metadata here
161         Box udta = createUdta(movie);
162         if (udta != null) {
163             movieBox.addBox(udta);
164         }
165         return movieBox;
166 
167     }

181         TrackHeaderBox tkhd = new TrackHeaderBox();
182         int flags = 0;
183         if (track.isEnabled()) {
184             flags += 1;
185         }
186 
187         if (track.isInMovie()) {

189         }
190

197         }

199

201         tkhd.setCreationTime(DateHelper.convert(track.getTrackMetaData().getCreationTime()));
202         // We need to take edit list box into account in trackheader duration
203         // but as long as I don't support edit list boxes it is sufficient to
204         // just translate media duration to movie timescale
205         tkhd.setDuration(getDuration(track) * getTimescale(movie) / track.getTrackMetaData().getTimescale());
206         tkhd.setHeight(track.getTrackMetaData().getHeight());
207         tkhd.setWidth(track.getTrackMetaData().getWidth());
208         tkhd.setLayer(track.getTrackMetaData().getLayer());
209         tkhd.setModificationTime(DateHelper.convert(new Date()));
210         tkhd.setTrackId(track.getTrackMetaData().getTrackId());
211         tkhd.setVolume(track.getTrackMetaData().getVolume());
212         if (tkhd.getCreationTime() >= 1l << 32 ||
213                 tkhd.getModificationTime() >= 1l << 32 ||
214                 tkhd.getDuration() >= 1l << 32) {
215             tkhd.setVersion(1);
216         }
217 
218         trackBox.addBox(tkhd);
219 
220 /*
221         EditBox edit = new EditBox();
222         EditListBox editListBox = new EditListBox();
223         editListBox.setEntries(Collections.singletonList(
224                 new EditListBox.Entry(editListBox, (long) (track.getTrackMetaData().getStartTime() * getTimescale(movie)), -1, 1)));
225         edit.addBox(editListBox);
226         trackBox.addBox(edit);
227 */
228 
229         MediaBox mdia = new MediaBox();
230         trackBox.addBox(mdia);
231         MediaHeaderBox mdhd = new MediaHeaderBox();
232         mdhd.setCreationTime(DateHelper.convert(track.getTrackMetaData().getCreationTime()));
233         mdhd.setDuration(getDuration(track));
234         mdhd.setTimescale(track.getTrackMetaData().getTimescale());
235         mdhd.setLanguage(track.getTrackMetaData().getLanguage());
236         mdia.addBox(mdhd);
237         HandlerBox hdlr = new HandlerBox();
238         mdia.addBox(hdlr);
239 
240         hdlr.setHandlerType(track.getHandler());
241 
242         MediaInformationBox minf = new MediaInformationBox();
243         minf.addBox(track.getMediaHeaderBox());
244 
245         // dinf: all these three boxes tell us is that the actual
246         // data is in the current file and not somewhere external
247         DataInformationBox dinf = new DataInformationBox();
248         DataReferenceBox dref = new DataReferenceBox();
249         dinf.addBox(dref);
250         DataEntryUrlBox url = new DataEntryUrlBox();
251         url.setFlags(1);
252         dref.addBox(url);
253         minf.addBox(dinf);
254         //
255 
256         SampleTableBox stbl = new SampleTableBox();
257 
258         stbl.addBox(track.getSampleDescriptionBox());
259 
260         List<TimeToSampleBox.Entry> decodingTimeToSampleEntries = track.getDecodingTimeEntries();
261         if (decodingTimeToSampleEntries != null && !track.getDecodingTimeEntries().isEmpty()) {
262             TimeToSampleBox stts = new TimeToSampleBox();
263             stts.setEntries(track.getDecodingTimeEntries());
264             stbl.addBox(stts);
265         }
266 
267         List<CompositionTimeToSample.Entry> compositionTimeToSampleEntries = track.getCompositionTimeEntries();
268         if (compositionTimeToSampleEntries != null && !compositionTimeToSampleEntries.isEmpty()) {
269             CompositionTimeToSample ctts = new CompositionTimeToSample();
270             ctts.setEntries(compositionTimeToSampleEntries);
271             stbl.addBox(ctts);
272         }
273 
274         long[] syncSamples = track.getSyncSamples();
275         if (syncSamples != null && syncSamples.length > 0) {
276             SyncSampleBox stss = new SyncSampleBox();
277             stss.setSampleNumber(syncSamples);
278             stbl.addBox(stss);
279         }
280 
281         if (track.getSampleDependencies() != null && !track.getSampleDependencies().isEmpty()) {
282             SampleDependencyTypeBox sdtp = new SampleDependencyTypeBox();
283             sdtp.setEntries(track.getSampleDependencies());
284             stbl.addBox(sdtp);
285         }
286         int[] tracksChunkSizes = chunks.get(track);
287 
288         SampleToChunkBox stsc = new SampleToChunkBox();
289         stsc.setEntries(new LinkedList<SampleToChunkBox.Entry>());
290         long lastChunkSize = Integer.MIN_VALUE; // to be sure the first chunks hasn't got the same size
291         for (int i = 0; i < tracksChunkSizes.length; i++) {
292             // The sample description index references the sample description box
293             // that describes the samples of this chunk. My Tracks cannot have more
294             // than one sample description box. Therefore 1 is always right
295             // the first chunk has the number '1'
296             if (lastChunkSize != tracksChunkSizes[i]) {
297                 stsc.getEntries().add(new SampleToChunkBox.Entry(i + 1, tracksChunkSizes[i], 1));
298                 lastChunkSize = tracksChunkSizes[i];
299             }
300         }
301         stbl.addBox(stsc);
302 
303         SampleSizeBox stsz = new SampleSizeBox();
304         stsz.setSampleSizes(track2SampleSizes.get(track));
305 
306         stbl.addBox(stsz);
307         // The ChunkOffsetBox we create here is just a stub
308         // since we haven't created the whole structure we can't tell where the
309         // first chunk starts (mdat box). So I just let the chunk offset
310         // start at zero and I will add the mdat offset later.
311         StaticChunkOffsetBox stco = new StaticChunkOffsetBox();
312         this.chunkOffsetBoxes.add(stco);
313         long offset = 0;
314         long[] chunkOffset = new long[tracksChunkSizes.length];
315         // all tracks have the same number of chunks
316         if (LOG.isLoggable(Level.FINE)) {
317             LOG.fine("Calculating chunk offsets for track_" + track.getTrackMetaData().getTrackId());
318         }
319 
320 
321         for (int i = 0; i < tracksChunkSizes.length; i++) {
322             // The filelayout will be:
323             // chunk_1_track_1,... ,chunk_1_track_n, chunk_2_track_1,... ,chunk_2_track_n, ... , chunk_m_track_1,... ,chunk_m_track_n
324             // calculating the offsets
325             if (LOG.isLoggable(Level.FINER)) {
326                 LOG.finer("Calculating chunk offsets for track_" + track.getTrackMetaData().getTrackId() + " chunk " + i);
327             }
328             for (Track current : movie.getTracks()) {
329                 if (LOG.isLoggable(Level.FINEST)) {
330                     LOG.finest("Adding offsets of track_" + current.getTrackMetaData().getTrackId());
331                 }
332                 int[] chunkSizes = chunks.get(current);
333                 long firstSampleOfChunk = 0;
334                 for (int j = 0; j < i; j++) {
335                     firstSampleOfChunk += chunkSizes[j];
336                 }
337                 if (current == track) {
338                     chunkOffset[i] = offset;
339                 }
340                 for (int j = l2i(firstSampleOfChunk); j < firstSampleOfChunk + chunkSizes[i]; j++) {
341                     offset += track2SampleSizes.get(current)[j];
342                 }
343             }
344         }
345         stco.setChunkOffsets(chunkOffset);
346         stbl.addBox(stco);
347         minf.addBox(stbl);
348         mdia.addBox(minf);
349 
350         return trackBox;
351     }

371         private InterleaveChunkMdat(Movie movie, Map<Track, int[]> chunks, long contentSize) {
372             this.contentSize = contentSize;
373             this.tracks = movie.getTracks();
374 
375             for (int i = 0; i < chunks.values().iterator().next().length; i++) {
376                 for (Track track : tracks) {
377 
378                     int[] chunkSizes = chunks.get(track);
379                     long firstSampleOfChunk = 0;
380                     for (int j = 0; j < i; j++) {
381                         firstSampleOfChunk += chunkSizes[j];
382                     }
383                     List<ByteBuffer> chunk = DefaultMp4Builder.this.track2Sample.get(track).subList(l2i(firstSampleOfChunk), l2i(firstSampleOfChunk + chunkSizes[i]));
384                     chunkList.add(chunk);
385                 }
386 
387             }
388 
389         }

391         public long getDataOffset() {
392             Box b = this;
393             long offset = 16;
394             while (b.getParent() != null) {
395                 for (Box box : b.getParent().getBoxes()) {
396                     if (b == box) {
397                         break;
398                     }
399                     offset += box.getSize();
400                 }
401                 b = b.getParent();
402             }
403             return offset;
404         }

420         public void getBox(WritableByteChannel writableByteChannel) throws IOException {
421             ByteBuffer bb = ByteBuffer.allocate(16);
422             long size = getSize();
423             if (isSmallBox(size)) {
424                 IsoTypeWriter.writeUInt32(bb, size);
425             } else {
426                 IsoTypeWriter.writeUInt32(bb, 1);
427             }
428             bb.put(IsoFile.fourCCtoBytes("mdat"));
429             if (isSmallBox(size)) {
430                 bb.put(new byte[8]);
431             } else {
432                 IsoTypeWriter.writeUInt64(bb, size);
433             }
434             bb.rewind();
435             writableByteChannel.write(bb);
436             for (List<ByteBuffer> samples : chunkList) {
437                 samples = unifyAdjacentBuffers(samples);
438                 for (ByteBuffer sample : samples) {
439                     writableByteChannel.write(sample);
440                 }
441             }
442         }

446     /**
447      * Gets the chunk sizes for the given track.
448      *
449      * @param track
450      * @param movie
451      * @return
452      */
453     int[] getChunkSizes(Track track, Movie movie) {
454 
455         long[] referenceChunkStarts = intersectionFinder.sampleNumbers(track, movie);
456         int[] chunkSizes = new int[referenceChunkStarts.length];
457 
458 
459         for (int i = 0; i < referenceChunkStarts.length; i++) {
460             long start = referenceChunkStarts[i] - 1;
461             long end;
462             if (referenceChunkStarts.length == i + 1) {
463                 end = track.getSamples().size();
464             } else {
465                 end = referenceChunkStarts[i + 1] - 1;
466             }
467 
468             chunkSizes[i] = l2i(end - start);
469             // The Stretch makes sure that there are as much audio and video chunks!
470         }
471         assert DefaultMp4Builder.this.track2Sample.get(track).size() == sum(chunkSizes) : "The number of samples and the sum of all chunk lengths must be equal";
472         return chunkSizes;
473 
474 
475     }

517     public List<ByteBuffer> unifyAdjacentBuffers(List<ByteBuffer> samples) {
518         ArrayList<ByteBuffer> nuSamples = new ArrayList<ByteBuffer>(samples.size());
519         samples = new ArrayList<ByteBuffer>(samples);
520         for (ByteBuffer buffer : samples) {
521             int lastIndex = nuSamples.size() - 1;
522             if (lastIndex >= 0 && buffer.hasArray() && nuSamples.get(lastIndex).hasArray() && buffer.array() == nuSamples.get(lastIndex).array() &&
523                     nuSamples.get(lastIndex).arrayOffset() + nuSamples.get(lastIndex).limit() == buffer.arrayOffset()) {
524                 ByteBuffer oldBuffer = nuSamples.remove(lastIndex);
525                 ByteBuffer nu = ByteBuffer.wrap(buffer.array(), oldBuffer.arrayOffset(), oldBuffer.limit() + buffer.limit()).slice();
526                 // We need to slice here since wrap([], offset, length) just sets position and not the arrayOffset.
527                 nuSamples.add(nu);
528             } else if (lastIndex >= 0 &&
529                     buffer instanceof MappedByteBuffer && nuSamples.get(lastIndex) instanceof MappedByteBuffer &&
530                     nuSamples.get(lastIndex).limit() == nuSamples.get(lastIndex).capacity() - buffer.capacity()) {
531                 // This can go wrong - but will it?
532                 ByteBuffer oldBuffer = nuSamples.get(lastIndex);
533                 oldBuffer.limit(buffer.limit() + oldBuffer.limit());
534             } else {
535                 nuSamples.add(buffer);
536             }
537         }
538         return nuSamples;
