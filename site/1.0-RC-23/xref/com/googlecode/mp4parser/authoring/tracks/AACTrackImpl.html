139     private void parse(ReadableByteChannel channel) throws IOException {
140 
141         stts = new LinkedList<TimeToSampleBox.Entry>();
142         samples = new LinkedList<ByteBuffer>();
143         firstHeader = readSamples(channel);
144 
145         double packetsPerSecond = (double) firstHeader.sampleRate / 1024.0;
146         double duration = samples.size() / packetsPerSecond;
147 
148         long dataSize = 0;
149         LinkedList<Integer> queue = new LinkedList<Integer>();
150         for (ByteBuffer sample : samples) {
151             int size = sample.remaining();
152             dataSize += size;
153             queue.add(size);
154             while (queue.size() > packetsPerSecond) {
155                 queue.pop();
156             }
157             if (queue.size() == (int) packetsPerSecond) {
158                 int currSize = 0;
159                 for (Integer aQueue : queue) {
160                     currSize += aQueue;
161                 }
162                 double currBitrate = 8.0 * currSize / queue.size() * packetsPerSecond;
163                 if (currBitrate > maxBitRate) {
164                     maxBitRate = (int) currBitrate;
165                 }
166             }
167         }
168 
169         avgBitRate = (int) (8 * dataSize / duration);
170 
171         bufferSizeDB = 1536; /* TODO: Calcultate this somehow! */
172 
173         sampleDescriptionBox = new SampleDescriptionBox();
174         AudioSampleEntry audioSampleEntry = new AudioSampleEntry("mp4a");
175         audioSampleEntry.setChannelCount(2);
176         audioSampleEntry.setSampleRate(firstHeader.sampleRate);
177         audioSampleEntry.setDataReferenceIndex(1);
178         audioSampleEntry.setSampleSize(16);
179 
180 
181         ESDescriptorBox esds = new ESDescriptorBox();
182         ESDescriptor descriptor = new ESDescriptor();
183         descriptor.setEsId(0);
184 
185         SLConfigDescriptor slConfigDescriptor = new SLConfigDescriptor();
186         slConfigDescriptor.setPredefined(2);
187         descriptor.setSlConfigDescriptor(slConfigDescriptor);
188 
189         DecoderConfigDescriptor decoderConfigDescriptor = new DecoderConfigDescriptor();
190         decoderConfigDescriptor.setObjectTypeIndication(0x40);
191         decoderConfigDescriptor.setStreamType(5);
192         decoderConfigDescriptor.setBufferSizeDB(bufferSizeDB);
193         decoderConfigDescriptor.setMaxBitRate(maxBitRate);
194         decoderConfigDescriptor.setAvgBitRate(avgBitRate);
195 
196         AudioSpecificConfig audioSpecificConfig = new AudioSpecificConfig();
197         audioSpecificConfig.setAudioObjectType(2); // AAC LC
198         audioSpecificConfig.setSamplingFrequencyIndex(firstHeader.sampleFrequencyIndex);
199         audioSpecificConfig.setChannelConfiguration(firstHeader.channelconfig);
200         decoderConfigDescriptor.setAudioSpecificInfo(audioSpecificConfig);
201 
202         descriptor.setDecoderConfigDescriptor(decoderConfigDescriptor);
203 
204         ByteBuffer data = descriptor.serialize();
205         esds.setData(data);
206         audioSampleEntry.addBox(esds);
207         sampleDescriptionBox.addBox(audioSampleEntry);
208 
209         trackMetaData.setCreationTime(new Date());
210         trackMetaData.setModificationTime(new Date());
211         trackMetaData.setLanguage(lang);
212         trackMetaData.setTimescale(firstHeader.sampleRate); // Audio tracks always use sampleRate as timescale
213     }

255     class AdtsHeader {
256         int getSize() {
257             return 7 + (protectionAbsent == 0 ? 2 : 0);
258         }
259 
260         int sampleFrequencyIndex;
261 
262         int mpegVersion;
263         int layer;
264         int protectionAbsent;
265         int profile;
266         int sampleRate;
267 
268         int channelconfig;
269         int original;
270         int home;
271         int copyrightedStream;
272         int copyrightStart;
273         int frameLength;
274         int bufferFullness;
275         int numAacFramesPerAdtsFrame;
276     }

278     private AdtsHeader readADTSHeader(ReadableByteChannel channel) throws IOException {
279         AdtsHeader hdr = new AdtsHeader();
280         ByteBuffer bb = ByteBuffer.allocate(7);
281         try {
282             ChannelHelper.readFully(channel, bb);
283         } catch (EOFException e) {
284             return null;
285         }
286         BitReaderBuffer brb = new BitReaderBuffer((ByteBuffer) bb.rewind());
287         int syncword = brb.readBits(12); // A
288         if (syncword != 0xfff) {
289             throw new IOException("Expected Start Word 0xfff");
290         }
291         hdr.mpegVersion = brb.readBits(1); // B
292         hdr.layer = brb.readBits(2); // C
293         hdr.protectionAbsent = brb.readBits(1); // D
294         hdr.profile = brb.readBits(2) + 1;  // E
295         //System.err.println(String.format("Profile %s", audioObjectTypes.get(hdr.profile)));
296         hdr.sampleFrequencyIndex = brb.readBits(4);
297         hdr.sampleRate = samplingFrequencyIndexMap.get(hdr.sampleFrequencyIndex); // F
298         brb.readBits(1); // G
299         hdr.channelconfig = brb.readBits(3); // H
300         hdr.original = brb.readBits(1); // I
301         hdr.home = brb.readBits(1); // J
302         hdr.copyrightedStream = brb.readBits(1); // K
303         hdr.copyrightStart = brb.readBits(1); // L
304         hdr.frameLength = brb.readBits(13); // M
305         //System.err.println(hdr.frameLength);
306         hdr.bufferFullness = brb.readBits(11); // 54
307         hdr.numAacFramesPerAdtsFrame = brb.readBits(2) + 1; // 56
308         if (hdr.numAacFramesPerAdtsFrame != 1) {
309             throw new IOException("This muxer can only work with 1 AAC frame per ADTS frame");
310         }
311         if (hdr.protectionAbsent == 0) {
312             channel.read(ByteBuffer.allocate(2));
313         }
314         return hdr;
315     }

317     private AdtsHeader readSamples(ReadableByteChannel channel) throws IOException {
318         AdtsHeader first = null;
319         AdtsHeader hdr;
320 
321         while ((hdr = readADTSHeader(channel)) != null) {
322             if (first == null) {
323                 first = hdr;
324             }
325             if (channel instanceof FileChannel) {
326                 ByteBuffer data = ((FileChannel) channel).map(FileChannel.MapMode.READ_ONLY, ((FileChannel) channel).position(), hdr.frameLength - hdr.getSize());
327                 samples.add(data);
328                 ((FileChannel) channel).position(((FileChannel) channel).position() + hdr.frameLength - hdr.getSize());
329                 data.rewind();
330             } else {
331                 ByteBuffer data = ByteBuffer.allocate(hdr.frameLength - hdr.getSize());
332                 channel.read(data);
333                 samples.add(data);
334                 data.rewind();
335             }
336             stts.add(new TimeToSampleBox.Entry(1, 1024));
337         }
338         return first;
339     }
