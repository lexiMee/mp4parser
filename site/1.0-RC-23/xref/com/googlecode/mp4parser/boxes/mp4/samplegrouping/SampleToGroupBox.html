29  /**
30   * <h1>4cc = "{@value #TYPE}"</h1>
31   * This table can be used to find the group that a sample belongs to and the associated description of that
32   * sample group. The table is compactly coded with each entry giving the index of the first sample of a run of
33   * samples with the same sample group descriptor. The sample group description ID is an index that refers to a
34   * SampleGroupDescription box, which contains entries describing the characteristics of each sample group.
35   * <p/>
36   * There may be multiple instances of this box if there is more than one sample grouping for the samples in a
37   * track. Each instance of the SampleToGroup box has a type code that distinguishes different sample
38   * groupings. Within a track, there shall be at most one instance of this box with a particular grouping type. The
39   * associated SampleGroupDescription shall indicate the same value for the grouping type.
40   * <p/>
41   * Version 1 of this box should only be used if a grouping type parameter is needed.
42   */
43  public class SampleToGroupBox extends AbstractFullBox {
44      public static final String TYPE = "sbgp";
45  
46  
47      private String groupingType;
48      private String groupingTypeParameter;
49  
50      List<Entry> entries = new LinkedList<Entry>();
51  
52      public SampleToGroupBox() {
53          super(TYPE);
54  
55      }
56  
57      @Override
58      protected long getContentSize() {
59          return this.getVersion() == 1 ? entries.size() * 8 + 16 : entries.size() * 8 + 12;
60      }
61  
62      @Override
63      protected void getContent(ByteBuffer byteBuffer) {
64          writeVersionAndFlags(byteBuffer);
65          byteBuffer.put(groupingType.getBytes());
66          if (this.getVersion() == 1) {
67              byteBuffer.put(groupingTypeParameter.getBytes());
68          }
69          IsoTypeWriter.writeUInt32(byteBuffer, entries.size());
70          for (Entry entry : entries) {
71              IsoTypeWriter.writeUInt32(byteBuffer, entry.getSampleCount());
72              IsoTypeWriter.writeUInt32(byteBuffer, entry.getGroupDescriptionIndex());
73          }
74  
75      }
76  
77      @Override
78      protected void _parseDetails(ByteBuffer content) {
79          parseVersionAndFlags(content);
80          groupingType = IsoTypeReader.read4cc(content);
81          if (this.getVersion() == 1) {
82              groupingTypeParameter = IsoTypeReader.read4cc(content);
83          }
84          long entryCount = IsoTypeReader.readUInt32(content);
85          while (entryCount-- > 0) {
86              entries.add(new Entry(l2i(IsoTypeReader.readUInt32(content)), l2i(IsoTypeReader.readUInt32(content))));
87          }
88      }

97          }

105         }

108             return groupDescriptionIndex;
109         }

112             this.groupDescriptionIndex = groupDescriptionIndex;
113         }
114 
115         @Override
116         public String toString() {
117             return "Entry{" +
118                     "sampleCount=" + sampleCount +
119                     ", groupDescriptionIndex=" + groupDescriptionIndex +
120                     '}';
121         }

124         public boolean equals(Object o) {
125             if (this == o) {
126                 return true;
127             }
128             if (o == null || getClass() != o.getClass()) {
129                 return false;
130             }
131

136             }

139             }

142         }

144         @Override
145         public int hashCode() {
146             int result = (int) (sampleCount ^ (sampleCount >>> 32));
147             result = 31 * result + groupDescriptionIndex;

149         }
150     }
