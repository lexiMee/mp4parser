41      public List<Long> getSyncSamples(SampleDependencyTypeBox sdtp) {
42          List<Long> result = new ArrayList<Long>();
43  
44          final List<SampleDependencyTypeBox.Entry> sampleEntries = sdtp.getEntries();
45          long i = 1;
46          for (SampleDependencyTypeBox.Entry sampleEntry : sampleEntries) {
47              if (sampleEntry.getSampleDependsOn() == 2) {
48                  result.add(i);
49              }
50              i++;
51          }
52  
53          return result;
54      }
55  
56      @DoNotParseDetail
57      public long getOffset() {
58          Box b = this;
59          long offset = 0;
60          while (b.getParent() != null) {
61              for (Box box : b.getParent().getBoxes()) {
62                  if (b == box) {
63                      break;
64                  }
65                  offset += box.getSize();
66              }
67              b = b.getParent();
68          }
69          return offset;
70      }

77      /**
78       * Returns the track numbers associated with this <code>MovieBox</code>.
79       *
80       * @return the tracknumbers (IDs) of the tracks in their order of appearance in the file
81       */
82  
83      public long[] getTrackNumbers() {
84  
85          List<TrackFragmentBox> trackBoxes = this.getBoxes(TrackFragmentBox.class, false);
86          long[] trackNumbers = new long[trackBoxes.size()];
87          for (int trackCounter = 0; trackCounter < trackBoxes.size(); trackCounter++) {
88              TrackFragmentBox trackBoxe = trackBoxes.get(trackCounter);
89              trackNumbers[trackCounter] = trackBoxe.getTrackFragmentHeaderBox().getTrackId();
90          }
91          return trackNumbers;
92      }
