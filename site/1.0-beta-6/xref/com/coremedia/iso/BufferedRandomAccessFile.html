77      /**
78       * Performs a seek and fills the buffer accordingly. *
79       */
80      void bufferSeek(long seekOffset) throws IOException {
81          flushBuffer();
82  
83          long newOffset = seekOffset - (seekOffset & (BUFFER_SIZE - 1L));
84          if (newOffset == bufferOffset) {
85              bufferPosition = (int) (seekOffset - bufferOffset);
86              return;
87          }
88  
89          bufferOffset = newOffset;
90          bufferLength = (int) min(BUFFER_SIZE, fileLength - bufferOffset);
91          if (bufferLength < 0)
92              bufferLength = 0;
93          bufferPosition = (int) (seekOffset - bufferOffset);
94  
95          // we always ask for an amount that should be exactly available.
96          raf.seek(bufferOffset);
97          raf.readFully(buffer, 0, bufferLength);
98  
99          // System.out.printf("%08x %08x %08x %08x\n", seekOffset, bufferOffset, bufferPosition, bufferLength);
100     }

155     public void readFully(byte[] b, int offset, int length) throws IOException {
156         while (length > 0) {
157             int bufferAvailable = bufferLength - bufferPosition;
158             int thiscopy = Math.min(bufferAvailable, length);
159             if (thiscopy == 0) {
160                 flushBuffer();
161 
162                 if (bufferOffset + bufferPosition >= fileLength)
163                     throw new EOFException("EOF");
164 
165                 bufferSeek(bufferOffset + bufferLength);
166                 continue;
167             }
168 
169             System.arraycopy(buffer, bufferPosition, b, offset, thiscopy);
170             bufferPosition += thiscopy;
171             offset += thiscopy;
172             length -= thiscopy;
173 
174         }
175     }

294     public void write(byte v) throws IOException {
295         bufferDirty = true;
296 
297         // they're doing a write within our current buffer.
298         if (bufferPosition < bufferLength) {
299             buffer[bufferPosition++] = v;
300             return;
301         }
302 
303         // they're increasing the size of the file, but it still fits inside our buffer
304         if (bufferLength < BUFFER_SIZE) {
305             buffer[bufferPosition++] = v;
306             bufferLength++;
307             fileLength++;
308             return;
309         }
310 
311         // they're doing a write, but we're out of buffer.
312         flushBuffer();
313         bufferSeek(bufferOffset + bufferPosition);
314         write(v);
315     }

334     public String readLine() throws IOException {
335         StringBuilder sb = null;
336 
337         while (true) {
338             int buffstart = bufferPosition;
339 
340             String piece = null;
341 
342             // suck as much out of this buffer as we can
343             while (bufferPosition < bufferLength) {
344                 char c = (char) (buffer[bufferPosition++] & 0xff);
345 
346                 if (c == '\n') {
347                     piece = new String(buffer, buffstart, bufferPosition - buffstart - 1);
348                     break;
349                 }
350 
351                 if (c == '\r') {
352                     piece = new String(buffer, buffstart, bufferPosition - buffstart - 1);
353 
354                     // this logic is untested.
355                     if (false && bufferPosition + bufferPosition < fileLength) {
356                         // consume \r\n if it appears
357                         if (peek() == '\n')
358                             read();
359                     }
360                     break;
361                 }
362             }
363 
364             // if a piece has been created, then we have found a newline
365             if (piece != null) {
366                 if (sb == null)
367                     return piece;
368 
369                 sb.append(piece);
370                 return sb.toString();
371             }
372 
373             piece = new String(buffer, buffstart, bufferPosition - buffstart);
374 
375             if (sb == null)
376                 sb = new StringBuilder();
377 
378             sb.append(piece);
379 
380             // EOF?
381             if (bufferOffset + bufferPosition >= fileLength) {
382                 // return the string so far...
383                 if (sb.length() > 0) {
384                     return sb.toString();
385                 } else
386                     return null; // EOF!
387             }
388 
389             bufferSeek(bufferOffset + bufferPosition);
390         }
