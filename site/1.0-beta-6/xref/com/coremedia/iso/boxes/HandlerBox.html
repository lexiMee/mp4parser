31   * This box within a Media Box declares the process by which the media-data in the track is presented,

66      private String name = null;

99      protected long getContentSize() {
100         if (zeroTerm) {
101             return 21 + utf8StringLengthInBytes(name);
102         } else {
103             return 20 + utf8StringLengthInBytes(name);
104         }
105 
106     }

108     public void parse(IsoBufferWrapper in, long size, BoxParser boxParser, Box lastMovieFragmentBox) throws IOException {
109         super.parse(in, size, boxParser, lastMovieFragmentBox);
110         in.readUInt32();
111         handlerType = IsoFile.bytesToFourCC(in.read(4));
112         a = in.readUInt32();
113         b = in.readUInt32();
114         c = in.readUInt32();
115         if ((int) (size - 24) > 0) {
116             name = in.readString((int) (size - 24));
117             if (name.endsWith("\0")) {
118                 name = name.substring(0, name.length() - 1);
119                 zeroTerm = true;
120             } else {
121                 zeroTerm = false;
122             }
123         } else {
124             zeroTerm = false; //No string at all, not even zero term char
125         }
126     }
127 
128     protected void getContent(IsoOutputStream isos) throws IOException {
129         isos.writeUInt32(0);
130         isos.write(IsoFile.fourCCtoBytes(handlerType));
131         isos.writeUInt32(a);
132         isos.writeUInt32(b);
133         isos.writeUInt32(c);
134         isos.writeStringNoTerm(name);
135         if (zeroTerm) {
136             isos.write(0);
137         }
138     }
139 
140     public String toString() {
141         return "HandlerBox[handlerType=" + getHandlerType() + ";name=" + getName() + "]";
142     }
