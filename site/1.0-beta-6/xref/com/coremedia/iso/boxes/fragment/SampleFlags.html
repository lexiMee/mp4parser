17  package com.coremedia.iso.boxes.fragment;

19  import com.coremedia.iso.IsoOutputStream;

21  import java.io.IOException;

23  /**
24   * bit(6) reserved=0;
25   * unsigned int(2) sample_depends_on;
26   * unsigned int(2) sample_is_depended_on;
27   * unsigned int(2) sample_has_redundancy;
28   * bit(3) sample_padding_value;
29   * bit(1) sample_is_difference_sample;
30   * // i.e. when 1 signals a non-key or non-sync sample
31   * unsigned int(16) sample_degradation_priority;
32   */
33  public class SampleFlags {
34    private int reserved;
35    private int sampleDependsOn;
36    private int sampleIsDependentOn;
37    private int sampleHasRedundancy;
38    private int samplePaddingValue;
39    private boolean sampleIsDifferenceSample;
40    private int sampleDegradationPriority;
41  
42    public SampleFlags(long flags) {
43      reserved = (int) (flags >> 26);
44      sampleDependsOn = (int) (flags >> 24) & 0x3;
45      sampleIsDependentOn = (int) (flags >> 22) & 0x3;
46      sampleHasRedundancy = (int) (flags >> 20) & 0x3;
47      samplePaddingValue = (int) (flags >> 17) & 0x7;
48      sampleIsDifferenceSample = ((flags >> 16) & 0x1) == 1;
49      sampleDegradationPriority = (int) flags & 0xFFFF;
50    }
51  
52      public int getReserved() {
53          return reserved;
54      }
55  
56      public int getSampleDependsOn() {
57          return sampleDependsOn;
58      }
59  
60      public int getSampleIsDependentOn() {
61          return sampleIsDependentOn;
62      }
63  
64      public int getSampleHasRedundancy() {
65          return sampleHasRedundancy;
66      }
67  
68      public int getSamplePaddingValue() {
69          return samplePaddingValue;
70      }
71  
72      public boolean isSampleIsDifferenceSample() {
73          return sampleIsDifferenceSample;
74      }
75  
76      public int getSampleDegradationPriority() {
77          return sampleDegradationPriority;
78      }
79  
80      public String toString() {
81          final StringBuilder sb = new StringBuilder();
82          sb.append("SampleFlags");
83          sb.append("{reserved=").append(reserved);
84          sb.append(", sampleDependsOn=").append(sampleDependsOn);
85          sb.append(", sampleIsDependentOn=").append(sampleIsDependentOn);
86          sb.append(", sampleHasRedundancy=").append(sampleHasRedundancy);
87          sb.append(", samplePaddingValue=").append(samplePaddingValue);
88          sb.append(", sampleIsDifferenceSample=").append(sampleIsDifferenceSample);
89          sb.append(", sampleDegradationPriority=").append(sampleDegradationPriority);
90          sb.append('}');
91          return sb.toString();
92      }
93  
94    public void getContent(IsoOutputStream os) throws IOException {
95      long flags = reserved << 26;
96      flags = flags | (sampleDependsOn << 24);
97      flags = flags | (sampleHasRedundancy << 22);
98      flags = flags | (samplePaddingValue << 19);
99      flags = flags | ((sampleIsDifferenceSample ? 1 : 0) << 18);
100     flags = flags | sampleDegradationPriority;
101     os.writeUInt32(flags);
102   }
