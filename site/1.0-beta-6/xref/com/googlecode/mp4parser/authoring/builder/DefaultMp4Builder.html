58      public IsoFile build(Movie movie) throws IOException {
59          LOG.info("Creating movie " + movie);
60          IsoFile isoFile = new IsoFile(new IsoBufferWrapperImpl(new byte[]{}));
61          isoFile.parse();
62          // ouch that is ugly but I don't know how to do it else
63          List<String> minorBrands = new LinkedList<String>();
64          minorBrands.add("isom");
65          minorBrands.add("iso2");
66          minorBrands.add("avc1");
67  
68          isoFile.addBox(new FileTypeBox("isom", 0, minorBrands));
69          isoFile.addBox(createMovieBox(movie));
70          Box mdat = new InterleaveChunkMdat(movie);
71          isoFile.addBox(mdat);
72          /*
73          dataOffset is where the first sample starts. Since we created the chunk offset boxes
74          without knowing this offset and temporarely
75           */
76          long dataOffset = mdat.calculateOffset() + 8;
77          for (StaticChunkOffsetBox chunkOffsetBox : chunkOffsetBoxes) {
78              long[] offsets = chunkOffsetBox.getChunkOffsets();
79              for (int i = 0; i < offsets.length; i++) {
80                  offsets[i] += dataOffset;
81              }
82          }
83  
84  
85          return isoFile;
86      }

88      private MovieBox createMovieBox(Movie movie) {
89          MovieBox movieBox = new MovieBox();
90          List<Box> movieBoxChildren = new LinkedList<Box>();
91          MovieHeaderBox mvhd = new MovieHeaderBox();
92          mvhd.setCreationTime(DateHelper.convert(new Date()));
93          mvhd.setModificationTime(DateHelper.convert(new Date()));
94  
95          long movieTimeScale = getTimescale(movie);
96          long duration = 0;
97  
98          for (Track track : movie.getTracks()) {
99              long tracksDuration = getDuration(track) * movieTimeScale / track.getTrackMetaData().getTimescale();
100             if (tracksDuration > duration) {
101                 duration = tracksDuration;
102             }
103 
104 
105         }
106 
107         mvhd.setDuration(duration);
108         mvhd.setTimescale(movieTimeScale);
109         // find the next available trackId
110         long nextTrackId = 0;
111         for (Track track : movie.getTracks()) {
112             nextTrackId = nextTrackId < track.getTrackMetaData().getTrackId() ? track.getTrackMetaData().getTrackId() : nextTrackId;
113         }
114         mvhd.setNextTrackId(++nextTrackId);
115         movieBoxChildren.add(mvhd);
116         for (Track track : movie.getTracks()) {
117             if (track.getType() != Track.Type.UNKNOWN) {
118                 movieBoxChildren.add(createTrackBox(track, movie));
119             }
120         }
121         // metadata here
122         movieBox.setBoxes(movieBoxChildren);
123         return movieBox;
124

255             stbl.addBox(sdtp);

257         long chunkSize[] = getChunkSizes(track, movie);
258         SampleToChunkBox stsc = new SampleToChunkBox();
259         stsc.setEntries(new LinkedList<SampleToChunkBox.Entry>());
260         long lastChunkSize = Integer.MIN_VALUE; // to be sure the first chunks hasn't got the same size
261         for (int i = 0; i < chunkSize.length; i++) {
262             // The sample description index references the sample description box
263             // that describes the samples of this chunk. My Tracks cannot have more
264             // than one sample description box. Therefore 1 is always right
265             // the first chunk has the number '1'
266             if (lastChunkSize != chunkSize[i]) {
267                 stsc.getEntries().add(new SampleToChunkBox.Entry(i + 1, chunkSize[i], 1));

270         }

282         // since we haven't created the whole structure we can't tell where the
283         // first chunk starts (mdat box). So I just let the chunk offset
284         // start at zero and I will add the mdat offset later.
285         StaticChunkOffsetBox stco = new StaticChunkOffsetBox();
286         this.chunkOffsetBoxes.add(stco);
287         long offset = 0;
288         long[] chunkOffset = new long[chunkSize.length];
289         // all tracks have the same number of chunks
290         LOG.fine("Calculating chunk offsets for track_" + track.getTrackMetaData().getTrackId());
291         for (int i = 0; i < chunkSize.length; i++) {
292             // The filelayout will be:
293             // chunk_1_track_1,... ,chunk_1_track_n, chunk_2_track_1,... ,chunk_2_track_n, ... , chunk_m_track_1,... ,chunk_m_track_n
294             // calculating the offsets
295             LOG.finer("Calculating chunk offsets for track_" + track.getTrackMetaData().getTrackId() + " chunk " + i);
296             for (Track current : movie.getTracks()) {
297                 LOG.finest("Adding offsets of track_" + current.getTrackMetaData().getTrackId());
298                 long[] chunkSizes = getChunkSizes(current, movie);
299                 long firstSampleOfChunk = 0;
300                 for (int j = 0; j < i; j++) {
301                     firstSampleOfChunk += chunkSizes[j];
302                 }
303                 if (current == track) {
304                     chunkOffset[i] = offset;
305                 }
306 
307                 for (long j = firstSampleOfChunk; j < firstSampleOfChunk + chunkSizes[i]; j++) {
308                     if (j > Integer.MAX_VALUE) {
309                         throw new InternalError("I cannot deal with a number of samples > Integer.MAX_VALUE");
310                     }
311 
312                     offset += current.getSamples().get((int) j).size();
313                 }
314             }
315         }
316         stco.setChunkOffsets(chunkOffset);
317         stbl.addBox(stco);
318         minf.addBox(stbl);
319         mdia.addBox(minf);
320 
321         return trackBox;
322     }

324     private static class InterleaveChunkMdat extends AbstractBox {
325         List<Track> tracks;
326         Map<Track, long[]> chunks = new HashMap<Track, long[]>();
327 
328         private InterleaveChunkMdat(Movie movie) {
329             super("mdat");
330             tracks = movie.getTracks();
331             for (Track track : movie.getTracks()) {
332                 chunks.put(track, getChunkSizes(track, movie));
333             }
334         }
335 
336         @Override
337         protected long getContentSize() {
338             long size = 0;
339             for (Track track : tracks) {
340                 for (IsoBufferWrapper sample : track.getSamples()) {
341                     size += sample.size();
342                 }
343             }
344             return size;
345         }

349             throw new InternalError("This box cannot be created by parsing");
350         }

352         @Override
353         protected void getContent(IsoOutputStream os) throws IOException {
354             long aaa = 0;
355             // all tracks have the same number of chunks
356             for (int i = 0; i < chunks.values().iterator().next().length; i++) {
357                 for (Track track : tracks) {
358 
359                     long[] chunkSizes = chunks.get(track);
360                     long firstSampleOfChunk = 0;
361                     for (int j = 0; j < i; j++) {
362                         firstSampleOfChunk += chunkSizes[j];
363                     }
364 
365                     for (long j = firstSampleOfChunk; j < firstSampleOfChunk + chunkSizes[i]; j++) {
366                         if (j > Integer.MAX_VALUE) {
367                             throw new InternalError("I cannot deal with a number of samples > Integer.MAX_VALUE");
368                         }
369 
370                         IsoBufferWrapper ibw = track.getSamples().get((int) j);
371                         while (ibw.remaining() >= 1024*1024) {
372                             os.write(ibw.read(1024*1024));
373                         }
374                         // it's safe to cast since there are less than 1024*1024 byte remaining
375                         os.write(ibw.read((int) ibw.remaining()));
376 
377                     }
378 
379                 }
380 
381             }
382             System.err.println(aaa);
383 
384         }

387     /**
388      * Gets the chunk sizes for the given track.
389      *
390      * @param track
391      * @param movie
392      * @return
393      */
394     static long[] getChunkSizes(Track track, Movie movie) {
395         Track referenceTrack = null;
396         long[] referenceChunkStarts = null;
397         long referenceSampleCount = 0;
398         long[] chunkSizes = null;
399         for (Track test : movie.getTracks()) {
400             if (test.getSyncSamples() != null && test.getSyncSamples().length > 0) {
401                 referenceTrack = test;
402                 referenceChunkStarts = test.getSyncSamples();
403                 referenceSampleCount = test.getSamples().size();
404                 chunkSizes = new long[referenceTrack.getSyncSamples().length];
405             }
406 
407         }
408         if (referenceTrack == null) {
409             referenceTrack = movie.getTracks().get(0);
410             referenceSampleCount = referenceTrack.getSamples().size();
411             int chunkCount = (int) (Math.ceil(getDuration(referenceTrack) / referenceTrack.getTrackMetaData().getTimescale()) / 2);
412             referenceChunkStarts = new long[chunkCount];
413             long chunkSize = referenceTrack.getSamples().size() / chunkCount;
414             for (int i = 0; i < referenceChunkStarts.length; i++) {
415                 referenceChunkStarts[i] = i * chunkSize;
416 
417             }
418 
419             chunkSizes = new long[chunkCount];
420         }
421 
422 
423         long sc = track.getSamples().size();
424         // Since the number of sample differs per track enormously 25 fps vs Audio for example
425         // we calculate the stretch. Stretch is the number of samples in current track that
426         // are needed for the time one sample in reference track is presented.
427         double stretch = (double) sc / referenceSampleCount;
428         for (int i = 0; i < chunkSizes.length; i++) {
429             long start = Math.round(stretch * ((referenceChunkStarts[i]) - 1));
430             long end = 0;
431             if (referenceChunkStarts.length == i + 1) {
432                 end = Math.round(stretch * (referenceSampleCount - 1));
433             } else {
434                 end = Math.round(stretch * ((referenceChunkStarts[i + 1] - 1)));
435             }
436 
437             chunkSizes[i] = end - start;
438             // The Stretch makes sure that there are as much audio and video chunks!
439         }
440         assert track.getSamples().size() == sum(chunkSizes) : "The number of samples and the sum of all chunk lengths must be equal";
441         return chunkSizes;
442 
443

463     public long getTimescale(Movie movie) {
464         long timescale = movie.getTracks().iterator().next().getTrackMetaData().getTimescale();
465         for (Track track : movie.getTracks()) {
466             timescale = gcd(track.getTrackMetaData().getTimescale(), timescale);
467         }
468         return timescale;

471     public static long gcd(long a, long b) {
472         if (b == 0) {
473             return a;
474         }
475         return gcd(b, a % b);
