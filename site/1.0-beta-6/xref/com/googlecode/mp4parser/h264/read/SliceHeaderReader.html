51      public SliceHeader read(NALUnit nalUnit, CAVLCReader reader)
52              throws IOException {
53  
54          SliceHeader sh = new SliceHeader();
55          sh.first_mb_in_slice = reader.readUE("SH: first_mb_in_slice");
56          int sh_type = reader.readUE("SH: slice_type");
57          sh.slice_type = SliceType.fromValue(sh_type % 5);
58          sh.slice_type_restr = (sh_type / 5) > 0;
59  
60          sh.pic_parameter_set_id = reader.readUE("SH: pic_parameter_set_id");
61  
62          PictureParameterSet pps = streamParams.getPPS(sh.pic_parameter_set_id);
63          SeqParameterSet sps = streamParams.getSPS(pps.seq_parameter_set_id);
64  
65          sh.pps = pps;
66          sh.sps = sps;
67  
68          sh.frame_num = reader.readU(sps.log2_max_frame_num_minus4 + 4,
69                  "SH: frame_num");
70          if (!sps.frame_mbs_only_flag) {
71              sh.field_pic_flag = reader.readBool("SH: field_pic_flag");
72              if (sh.field_pic_flag) {
73                  sh.bottom_field_flag = reader.readBool("SH: bottom_field_flag");
74              }
75          }
76          if (nalUnit.type == NALUnitType.IDR_SLICE) {
77              sh.idr_pic_id = reader.readUE("SH: idr_pic_id");
78          }
79          if (sps.pic_order_cnt_type == 0) {
80              sh.pic_order_cnt_lsb = reader.readU(
81                      sps.log2_max_pic_order_cnt_lsb_minus4 + 4,
82                      "SH: pic_order_cnt_lsb");
83              if (pps.pic_order_present_flag && !sps.field_pic_flag) {
84                  sh.delta_pic_order_cnt_bottom = reader
85                          .readSE("SH: delta_pic_order_cnt_bottom");
86              }
87          }
88          sh.delta_pic_order_cnt = new int[2];
89          if (sps.pic_order_cnt_type == 1
90                  && !sps.delta_pic_order_always_zero_flag) {
91              sh.delta_pic_order_cnt[0] = reader
92                      .readSE("SH: delta_pic_order_cnt[0]");
93              if (pps.pic_order_present_flag && !sps.field_pic_flag)
94                  sh.delta_pic_order_cnt[1] = reader
95                          .readSE("SH: delta_pic_order_cnt[1]");
96          }
97          if (pps.redundant_pic_cnt_present_flag) {
98              sh.redundant_pic_cnt = reader.readUE("SH: redundant_pic_cnt");
99          }
100         if (sh.slice_type == SliceType.B) {
101             sh.direct_spatial_mv_pred_flag = reader
102                     .readBool("SH: direct_spatial_mv_pred_flag");
103         }
104         if (sh.slice_type == SliceType.P || sh.slice_type == SliceType.SP
105                 || sh.slice_type == SliceType.B) {
106             sh.num_ref_idx_active_override_flag = reader
107                     .readBool("SH: num_ref_idx_active_override_flag");
108             if (sh.num_ref_idx_active_override_flag) {
109                 sh.num_ref_idx_l0_active_minus1 = reader
110                         .readUE("SH: num_ref_idx_l0_active_minus1");
111                 if (sh.slice_type == SliceType.B) {
112                     sh.num_ref_idx_l1_active_minus1 = reader
113                             .readUE("SH: num_ref_idx_l1_active_minus1");
114                 }
115             }
116         }
117         readRefPicListReordering(sh, reader);
118         if ((sps.weighted_pred_flag && (sh.slice_type == SliceType.P || sh.slice_type == SliceType.SP))
119                 || (sps.weighted_bipred_idc == 1 && sh.slice_type == SliceType.B))
120             readPredWeightTable(sps, pps, sh, reader);
121         if (nalUnit.nal_ref_idc != 0)
122             readDecoderPicMarking(nalUnit, sh, reader);
123         if (sps.entropy_coding_mode_flag && sh.slice_type != SliceType.I
124                 && sh.slice_type != SliceType.SI) {
125             sh.cabac_init_idc = reader.readUE("SH: cabac_init_idc");
126         }
127         sh.slice_qp_delta = reader.readSE("SH: slice_qp_delta");
128         if (sh.slice_type == SliceType.SP || sh.slice_type == SliceType.SI) {
129             if (sh.slice_type == SliceType.SP) {
130                 sh.sp_for_switch_flag = reader
131                         .readBool("SH: sp_for_switch_flag");
132             }
133             sh.slice_qs_delta = reader.readSE("SH: slice_qs_delta");
134         }
135         if (pps.deblocking_filter_control_present_flag) {
136             sh.disable_deblocking_filter_idc = reader
137                     .readUE("SH: disable_deblocking_filter_idc");
138             if (sh.disable_deblocking_filter_idc != 1) {
139                 sh.slice_alpha_c0_offset_div2 = reader
140                         .readSE("SH: slice_alpha_c0_offset_div2");
141                 sh.slice_beta_offset_div2 = reader
142                         .readSE("SH: slice_beta_offset_div2");
143             }
144         }
145         if (pps.num_slice_groups_minus1 > 0 && pps.slice_group_map_type >= 3
146                 && pps.slice_group_map_type <= 5) {
147             int len = (sps.pic_height_in_map_units_minus1 + 1)
148                     * (sps.pic_width_in_mbs_minus1 + 1)
149                     / (pps.slice_group_change_rate_minus1 + 1);
150             if (((sps.pic_height_in_map_units_minus1 + 1) * (sps.pic_width_in_mbs_minus1 + 1))
151                     % (pps.slice_group_change_rate_minus1 + 1) > 0)
152                 len += 1;
153 
154             len = CeilLog2(len + 1);
155             sh.slice_group_change_cycle = reader.readU(len,
156                     "SH: slice_group_change_cycle");
157         }
158 
159         return sh;
160     }

162     private static int CeilLog2(int uiVal) {
163         int uiTmp = uiVal - 1;
164         int uiRet = 0;
165 
166         while (uiTmp != 0) {
167             uiTmp >>= 1;
168             uiRet++;
169         }
170         return uiRet;
171     }

173     private static void readDecoderPicMarking(NALUnit nalUnit, SliceHeader sh,
174                                               CAVLCReader reader) throws IOException {
175         if (nalUnit.type == NALUnitType.IDR_SLICE) {
176             boolean no_output_of_prior_pics_flag = reader
177                     .readBool("SH: no_output_of_prior_pics_flag");
178             boolean long_term_reference_flag = reader
179                     .readBool("SH: long_term_reference_flag");
180             sh.refPicMarkingIDR = new RefPicMarkingIDR(
181                     no_output_of_prior_pics_flag, long_term_reference_flag);
182         } else {
183             boolean adaptive_ref_pic_marking_mode_flag = reader
184                     .readBool("SH: adaptive_ref_pic_marking_mode_flag");
185             if (adaptive_ref_pic_marking_mode_flag) {
186                 ArrayList<RefPicMarking.Instruction> mmops = new ArrayList<RefPicMarking.Instruction>();
187                 int memory_management_control_operation;
188                 do {
189                     memory_management_control_operation = reader
190                             .readUE("SH: memory_management_control_operation");
191 
192                     RefPicMarking.Instruction instr = null;
193 
194                     switch (memory_management_control_operation) {
195                         case 1:
196                             instr = new RefPicMarking.Instruction(
197                                     RefPicMarking.InstrType.REMOVE_SHORT,
198                                     reader
199                                             .readUE("SH: difference_of_pic_nums_minus1") + 1,
200                                     0);
201                             break;
202                         case 2:
203                             instr = new RefPicMarking.Instruction(
204                                     RefPicMarking.InstrType.REMOVE_LONG, reader
205                                     .readUE("SH: long_term_pic_num"), 0);
206                             break;
207                         case 3:
208                             instr = new RefPicMarking.Instruction(
209                                     RefPicMarking.InstrType.CONVERT_INTO_LONG,
210                                     reader
211                                             .readUE("SH: difference_of_pic_nums_minus1") + 1,
212                                     reader.readUE("SH: long_term_frame_idx"));
213                             break;
214                         case 4:
215                             instr = new RefPicMarking.Instruction(
216                                     RefPicMarking.InstrType.TRUNK_LONG,
217                                     reader
218                                             .readUE("SH: max_long_term_frame_idx_plus1"),
219                                     0);
220                             break;
221                         case 5:
222                             instr = new RefPicMarking.Instruction(RefPicMarking.InstrType.CLEAR,
223                                     0, 0);
224                             break;
225                         case 6:
226                             instr = new RefPicMarking.Instruction(
227                                     RefPicMarking.InstrType.MARK_LONG, reader
228                                     .readUE("SH: long_term_frame_idx"), 0);
229                             break;
230                     }
231                     if (instr != null)
232                         mmops.add(instr);
233                 } while (memory_management_control_operation != 0);
234                 sh.refPicMarkingNonIDR = new RefPicMarking(mmops
235                         .toArray(new RefPicMarking.Instruction[]{}));
236             }
237         }
238     }

240     private static void readPredWeightTable(SeqParameterSet sps,
241                                             PictureParameterSet pps, SliceHeader sh, CAVLCReader reader)
242             throws IOException {
243         sh.pred_weight_table.luma_log2_weight_denom = reader
244                 .readUE("SH: luma_log2_weight_denom");
245         if (sps.chroma_format_idc != ChromaFormat.MONOCHROME) {
246             sh.pred_weight_table.chroma_log2_weight_denom = reader
247                     .readUE("SH: chroma_log2_weight_denom");
248         }
249         sh.pred_weight_table.luma_offset_weight_l0 = new PredictionWeightTable.OffsetWeight[pps.num_ref_idx_l0_active_minus1];
250         sh.pred_weight_table.chroma_offset_weight_l0 = new PredictionWeightTable.OffsetWeight[pps.num_ref_idx_l0_active_minus1][];
251         for (int i = 0; i <= pps.num_ref_idx_l0_active_minus1; i++) {
252             boolean luma_weight_l0_flag = reader
253                     .readBool("SH: luma_weight_l0_flag");
254             if (luma_weight_l0_flag) {
255                 PredictionWeightTable.OffsetWeight ow = new PredictionWeightTable.OffsetWeight();
256                 ow.weight = reader.readSE("SH: weight");
257                 ow.offset = reader.readSE("SH: offset");
258 
259                 sh.pred_weight_table.luma_offset_weight_l0[i] = ow;
260             }
261             if (sps.chroma_format_idc != ChromaFormat.MONOCHROME) {
262                 boolean chroma_weight_l0_flag = reader
263                         .readBool("SH: chroma_weight_l0_flag");
264                 if (chroma_weight_l0_flag)
265                     for (int j = 0; j < 2; j++) {
266                         PredictionWeightTable.OffsetWeight ow = new PredictionWeightTable.OffsetWeight();
267                         ow.weight = reader.readSE("SH: weight");
268                         ow.offset = reader.readSE("SH: offset");
269                         sh.pred_weight_table.chroma_offset_weight_l0[i][j] = ow;
270                     }
271             }
272         }
273         if (sh.slice_type == SliceType.B) {
274             sh.pred_weight_table.luma_offset_weight_l1 = new PredictionWeightTable.OffsetWeight[pps.num_ref_idx_l1_active_minus1];
275             sh.pred_weight_table.chroma_offset_weight_l1 = new PredictionWeightTable.OffsetWeight[pps.num_ref_idx_l1_active_minus1][];
276             for (int i = 0; i <= pps.num_ref_idx_l1_active_minus1; i++) {
277                 boolean luma_weight_l1_flag = reader
278                         .readBool("SH: luma_weight_l1_flag");
279                 if (luma_weight_l1_flag) {
280                     PredictionWeightTable.OffsetWeight ow = new PredictionWeightTable.OffsetWeight();
281                     ow.weight = reader.readSE("SH: weight");
282                     ow.offset = reader.readSE("SH: offset");
283                     sh.pred_weight_table.luma_offset_weight_l1[i] = ow;
284                 }
285                 if (sps.chroma_format_idc != ChromaFormat.MONOCHROME) {
286                     boolean chroma_weight_l1_flag = reader
287                             .readBool("SH: chroma_weight_l1_flag");
288                     if (chroma_weight_l1_flag) {
289                         for (int j = 0; j < 2; j++) {
290                             PredictionWeightTable.OffsetWeight ow = new PredictionWeightTable.OffsetWeight();
291                             ow.weight = reader.readSE("SH: weight");
292                             ow.offset = reader.readSE("SH: offset");
293                             sh.pred_weight_table.chroma_offset_weight_l1[i][j] = ow;
294                         }
295                     }
296                 }
297             }
298         }
299     }

301     private static void readRefPicListReordering(SliceHeader sh,
302                                                  CAVLCReader reader) throws IOException {
303         if (sh.slice_type != SliceType.I && sh.slice_type != SliceType.SI) {
304             boolean ref_pic_list_reordering_flag_l0 = reader
305                     .readBool("SH: ref_pic_list_reordering_flag_l0");
306             if (ref_pic_list_reordering_flag_l0) {
307                 sh.refPicReordering = readReorderingEntries(reader);
308             }
309         }
310         if (sh.slice_type == SliceType.B) {
311             boolean ref_pic_list_reordering_flag_l1 = reader
312                     .readBool("SH: ref_pic_list_reordering_flag_l1");
313             if (ref_pic_list_reordering_flag_l1) {
314                 readReorderingEntries(reader);
315             }
316         }
317     }

319     private static RefPicReordering readReorderingEntries(CAVLCReader reader)
320             throws IOException {
321         ArrayList<RefPicReordering.ReorderOp> reordering = new ArrayList<RefPicReordering.ReorderOp>();
322         int reordering_of_pic_nums_idc;
323         do {
324             reordering_of_pic_nums_idc = reader
325                     .readUE("SH: reordering_of_pic_nums_idc");
326             switch (reordering_of_pic_nums_idc) {
327                 case 0:
328                     reordering.add(new RefPicReordering.ReorderOp(RefPicReordering.InstrType.BACKWARD, reader
329                             .readUE("SH: abs_diff_pic_num_minus1") + 1));
330                     break;
331                 case 1:
332                     reordering.add(new RefPicReordering.ReorderOp(RefPicReordering.InstrType.FORWARD, reader
333                             .readUE("SH: abs_diff_pic_num_minus1") + 1));
334                     break;
335                 case 2:
336                     reordering.add(new RefPicReordering.ReorderOp(RefPicReordering.InstrType.LONG_TERM, reader
337                             .readUE("SH: long_term_pic_num")));
338                     break;
339             }
340         } while (reordering_of_pic_nums_idc != 3);
341         return new RefPicReordering(reordering.toArray(new RefPicReordering.ReorderOp[]{}));
342     }
