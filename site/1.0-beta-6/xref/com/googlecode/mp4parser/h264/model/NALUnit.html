21  package com.googlecode.mp4parser.h264.model;
22  
23  import com.coremedia.iso.IsoBufferWrapper;
24  
25  import java.io.IOException;
26  import java.io.OutputStream;
27  
28  /**
29   * Network abstraction layer (NAL) unit
30   *
31   * @author Stanislav Vitvitskiy
32   */
33  public class NALUnit {
34  
35      public NALUnitType type;
36      public int nal_ref_idc;
37  
38      public NALUnit(NALUnitType type, int nal_ref_idc) {
39          this.type = type;
40          this.nal_ref_idc = nal_ref_idc;
41      }
42  
43      public static NALUnit read(IsoBufferWrapper is) throws IOException {
44          int nalu = is.read();
45          int nal_ref_idc = (nalu >> 5) & 0x3;
46          int nb = nalu & 0x1f;
47  
48          NALUnitType type = NALUnitType.fromValue(nb);
49          return new NALUnit(type, nal_ref_idc);
50      }
51  
52      public void write(OutputStream out) throws IOException {
53          int nalu = type.getValue() | (nal_ref_idc << 5);
54          out.write(nalu);
55      }
56  
57      @Override
58      public String toString() {
59          return "NALUnit{" +
60                  "type=" + type +
61                  ", nal_ref_idc=" + nal_ref_idc +
62                  '}';
63      }
64  }
