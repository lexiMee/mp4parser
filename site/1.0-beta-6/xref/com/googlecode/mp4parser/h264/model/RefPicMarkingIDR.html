24   * Reference picture marking used for IDR frames

28  public class RefPicMarkingIDR {

30      boolean useForlongTerm;

33          this.discardDecodedPics = discardDecodedPics;
34          this.useForlongTerm = useForlongTerm;
35      }
36  
37      public boolean isDiscardDecodedPics() {

39      }
40  
41      public boolean isUseForlongTerm() {

43      }
44

49                  ", useForlongTerm=" + useForlongTerm +
50                  '}';
51      }
52  }
