334     private void parse(IsoBufferWrapper in, int maxLength) throws IOException {
335         final long position = in.position();
336         endPos = position + maxLength;
337         configBytes = in.read(maxLength);
338         in.position(position);
339 
340         audioObjectType = getAudioObjectType(in);
341         samplingFrequencyIndex = in.readBits(4);
342 
343         if (samplingFrequencyIndex == 0xf) {
344             samplingFrequency = in.readBits(24);
345         }
346 
347         channelConfiguration = in.readBits(4);
348 
349         if (audioObjectType == 5 ||
350                 audioObjectType == 29) {
351             extensionAudioObjectType = 5;
352             sbrPresentFlag = 1;
353             if (audioObjectType == 29) {
354                 psPresentFlag = 1;
355             }
356             extensionSamplingFrequencyIndex = in.readBits(4);
357             if (extensionSamplingFrequencyIndex == 0xf)
358                 extensionSamplingFrequency = in.readBits(24);
359             audioObjectType = getAudioObjectType(in);
360             if (audioObjectType == 22)
361                 extensionChannelConfiguration = in.readBits(4);
362         } else {
363             extensionAudioObjectType = 0;
364         }
365 
366         switch (audioObjectType) {
367             case 1:
368             case 2:
369             case 3:
370             case 4:
371             case 6:
372             case 7:
373             case 17:
374             case 19:
375             case 20:
376             case 21:
377             case 22:
378             case 23:
379                 parseGaSpecificConfig(samplingFrequencyIndex, channelConfiguration, audioObjectType, in);
380                 //GASpecificConfig();
381                 break;
382             case 8:
383                 throw new UnsupportedOperationException("can't parse CelpSpecificConfig yet");
384                 //CelpSpecificConfig();
385                 //break;
386             case 9:
387                 throw new UnsupportedOperationException("can't parse HvxcSpecificConfig yet");
388                 //HvxcSpecificConfig();
389                 //break;
390             case 12:
391                 throw new UnsupportedOperationException("can't parse TTSSpecificConfig yet");
392                 //TTSSpecificConfig();
393                 //break;
394             case 13:
395             case 14:
396             case 15:
397             case 16:
398                 throw new UnsupportedOperationException("can't parse StructuredAudioSpecificConfig yet");
399                 //StructuredAudioSpecificConfig();
400                 //break;
401             case 24:
402                 throw new UnsupportedOperationException("can't parse ErrorResilientCelpSpecificConfig yet");
403                 //ErrorResilientCelpSpecificConfig();
404                 //break;
405             case 25:
406                 throw new UnsupportedOperationException("can't parse ErrorResilientHvxcSpecificConfig yet");
407                 //ErrorResilientHvxcSpecificConfig();
408                 //break;
409             case 26:
410             case 27:
411                 parseParametricSpecificConfig(samplingFrequencyIndex, channelConfiguration, audioObjectType, in);
412                 //ParametricSpecificConfig();
413                 break;
414             case 28:
415                 throw new UnsupportedOperationException("can't parse SSCSpecificConfig yet");
416                 //SSCSpecificConfig();
417                 //break;
418             case 30:
419                 sacPayloadEmbedding = in.readBits(1);
420                 throw new UnsupportedOperationException("can't parse SpatialSpecificConfig yet");
421                 //SpatialSpecificConfig();
422                 //break;
423             case 32:
424             case 33:
425             case 34:
426                 throw new UnsupportedOperationException("can't parse MPEG_1_2_SpecificConfig yet");
427                 //MPEG_1_2_SpecificConfig();
428                 //break;
429             case 35:
430                 throw new UnsupportedOperationException("can't parse DSTSpecificConfig yet");
431                 //DSTSpecificConfig();
432                 //break;
433             case 36:
434                 fillBits = in.readBits(5);
435                 throw new UnsupportedOperationException("can't parse ALSSpecificConfig yet");
436                 //ALSSpecificConfig();
437                 //break;
438             case 37:
439             case 38:
440                 throw new UnsupportedOperationException("can't parse SLSSpecificConfig yet");
441                 //SLSSpecificConfig();
442                 //break;
443             case 39:
444                 throw new UnsupportedOperationException("can't parse ELDSpecificConfig yet");
445                 //ELDSpecificConfig(channelConfiguration);
446                 //break;
447             case 40:
448             case 41:
449                 throw new UnsupportedOperationException("can't parse SymbolicMusicSpecificConfig yet");
450                 //SymbolicMusicSpecificConfig();
451                 //break;
452             default:
453                 /* reserved */
454         }
455 
456         switch (audioObjectType) {
457             case 17:
458             case 19:
459             case 20:
460             case 21:
461             case 22:
462             case 23:
463             case 24:
464             case 25:
465             case 26:
466             case 27:
467             case 39:
468                 epConfig = in.readBits(2);
469                 if (epConfig == 2 || epConfig == 3) {
470                     throw new UnsupportedOperationException("can't parse ErrorProtectionSpecificConfig yet");
471                     //ErrorProtectionSpecificConfig();
472                 }
473                 if (epConfig == 3) {
474                     directMapping = in.readBits(1);
475                     if (directMapping == 0) {
476                         /* tbd */
477                     }
478                 }
479         }
480 
481         if (extensionAudioObjectType != 5 && bits_to_decode(in) >= 16) {
482             syncExtensionType = in.readBits(11);
483             if (syncExtensionType == 0x2b7) {
484                 extensionAudioObjectType = getAudioObjectType(in);
485                 if (extensionAudioObjectType == 5) {
486                     sbrPresentFlag = in.readBits(1);
487                     if (sbrPresentFlag == 1) {
488                         extensionSamplingFrequencyIndex = in.readBits(4);
489                         if (extensionSamplingFrequencyIndex == 0xf) {
490                             extensionSamplingFrequency = in.readBits(24);
491                         }
492                         if (bits_to_decode(in) >= 12) {
493                             syncExtensionType = in.readBits(11); //10101001000
494                             if (syncExtensionType == 0x548) {
495                                 psPresentFlag = in.readBits(1);
496                             }
497                         }
498                     }
499                 }
500                 if (extensionAudioObjectType == 22) {
501                     sbrPresentFlag = in.readBits(1);
502                     if (sbrPresentFlag == 1) {
503                         extensionSamplingFrequencyIndex = in.readBits(4);
504                         if (extensionSamplingFrequencyIndex == 0xf) {
505                             extensionSamplingFrequency = in.readBits(24);
506                         }
507                     }
508                     extensionChannelConfiguration = in.readBits(4);
509                 }
510             }
511         }
512     }

514     private int bits_to_decode(IsoBufferWrapper in) throws IOException {
515         return (int) ((endPos - in.position()) * 8 + in.getReadBitsRemaining());
516     }

518     private int getAudioObjectType(IsoBufferWrapper in) throws IOException {
519         int audioObjectType = in.readBits(5);
520         if (audioObjectType == 31) {
521             audioObjectType = 32 + in.readBits(6);
522         }
523         return audioObjectType;
524     }

564     private void parseParametricSpecificConfig(int samplingFrequencyIndex, int channelConfiguration, int audioObjectType, IsoBufferWrapper in) throws IOException {
565         /*
566         ParametricSpecificConfig() {
567             isBaseLayer; 1 uimsbf
568             if (isBaseLayer) {
569                 PARAconfig();
570             } else {
571                 HILNenexConfig();
572             }
573         }
574         */
575         isBaseLayer = in.readBits(1);
576         if (isBaseLayer == 1) {
577             parseParaConfig(samplingFrequencyIndex, channelConfiguration, audioObjectType, in);
578         } else {
579             parseHilnEnexConfig(samplingFrequencyIndex, channelConfiguration, audioObjectType, in);
580         }

583     private void parseParaConfig(int samplingFrequencyIndex, int channelConfiguration, int audioObjectType, IsoBufferWrapper in) throws IOException {
584         /*
585         PARAconfig()
586         {
587             PARAmode; 2 uimsbf
588             if (PARAmode != 1) {
589                 ErHVXCconfig();
590             }
591             if (PARAmode != 0) {
592                 HILNconfig();
593             }
594             PARAextensionFlag; 1 uimsbf
595             if (PARAextensionFlag) {
596                 // to be defined in MPEG-4 Phase 3
597             }
598         }
599         */
600         paraMode = in.readBits(2);
601 
602         if (paraMode != 1) {
603             parseErHvxcConfig(samplingFrequencyIndex, channelConfiguration, audioObjectType, in);
604         }
605         if (paraMode != 0) {
606             parseHilnConfig(samplingFrequencyIndex, channelConfiguration, audioObjectType, in);
607         }
608 
609         paraExtensionFlag = in.readBits(1);
610         parametricSpecificConfig = true;
611     }

613     private void parseErHvxcConfig(int samplingFrequencyIndex, int channelConfiguration, int audioObjectType, IsoBufferWrapper in) throws IOException {
614         /*
615         ErHVXCconfig()
616         {
617             HVXCvarMode; 1 uimsbf
618                 HVXCrateMode; 2 uimsbf
619                 extensionFlag; 1 uimsbf
620             if (extensionFlag) {
621                 var_ScalableFlag; 1 uimsbf
622             }
623         }
624         */
625         hvxcVarMode = in.readBits(1);
626         hvxcRateMode = in.readBits(2);
627         erHvxcExtensionFlag = in.readBits(1);
628 
629         if (erHvxcExtensionFlag == 1) {
630             var_ScalableFlag = in.readBits(1);
631         }
632     }

635         /*

638             HILNquantMode; 1 uimsbf

640             HILNsampleRateCode; 4 uimsbf
641             HILNframeLength; 12 uimsbf
642             HILNcontMode; 2 uimsbf
643         }

645         hilnQuantMode = in.readBits(1);

647         hilnSampleRateCode = in.readBits(4);
648         hilnFrameLength = in.readBits(12);
